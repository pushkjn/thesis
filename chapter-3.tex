\chapter[Гармонические функции]{ГАРМОНИЧЕСКИЕ ФУНКЦИИ}
\section{Введение в обобщённые функции. Фундаментальное решение оператора Лапласа}
\markright{\thesection\ Фундаментальное решение лапласиана}

Дадим многомерное обобщение хорошо известной формулы интегрирования по частям из математического анализа 
\begin{equation*}
\int \limits_{a}^{b} u v' \, dx = \left. uv \right|_{a}^{b} - \int \limits_{a}^{b} u' v \, dx, 
\end{equation*}
где $u, v \in C^{1}[a,b]$. Пусть $\Omega$ --- ограниченная область с границей класса $C^1$, $u,v \in C^{1} \left( \overline{\Omega} \right)$, $\vec{\nu}$ --- непрерывное поле единичных внешних нормалей к $\partial \Omega$. Тогда для любого индекса $j=1,\ldots,n$
\begin{equation}
\label{2.1}
 \int \limits_{\Omega} u \frac{\partial v}{\partial x_j} \, dx = \int \limits_{\partial \Omega} u v \nu_{j} \, ds - \int \limits_{\Omega} \frac{\partial u}{\partial x_j} v \, dx,
 \end{equation} 
где $\nu_{j}$ --- $j$-ая координата нормального вектора, $ds$ --- элемент гиперплощади. Формула \eqref{2.1} легко следует из формулы Гаусса--Остроградского $\int_{\partial \Omega} \vec{a} \cdot \vec{\nu}\, ds = \int_{\Omega} \mathrm{div}\, \vec{a} \, dx$, где $\vec{a} = a_{1} \vec{e}_{1} + \ldots + a_{n} \vec{e}_{n} $, $a_{j} \in C^{1} \left( \overline{\Omega} \right)$, $j=1,\ldots,n$, если последнюю применить к векторному полю $\vec{a} = u v \vec{e}_{j}$. Из формулы интегрирования по частям следуют две формулы Грина для оператора Лапласа. Для функций $u,v \in C^{2} \left( \overline{\Omega} \right)$ первая формула Грина
\begin{equation}
\int \limits_{\Omega} u \Delta v \, dx = \int \limits_{\partial \Omega} u \frac{\partial v}{\partial \vec{\nu}} \, ds - \int \limits_{\Omega} \nabla u \cdot \nabla v \, dx, \label{2.2}
\end{equation}
где $\frac{\partial v}{\partial \vec{\nu}} = \frac{\partial v}{\partial x_1} \nu_1 + \ldots + \frac{\partial v}{\partial x_n} \nu_n$ --- нормальная производная, получается суммированием по  $j=1,\ldots,n$ соотношений
\begin{equation*}
\int \limits_{\Omega} u \frac{\partial^2 v}{\partial x_j^2} \, dx = \int \limits_{\partial \Omega} u \frac{\partial v}{\partial x_j} \nu_{j} \, ds - \int \limits_{\Omega} \frac{\partial u}{\partial x_j} \frac{\partial v}{\partial x_j} \, dx.
\end{equation*}
Если в формуле \eqref{2.2} поменять функции $u$ и $v$ местами, а затем вычесть две первые формулы Грина друг из друга, получится вторая формула Грина для оператора Лапласа:
\begin{equation}
 \int \limits_{\Omega} u \Delta v \, dx = \int \limits_{\partial \Omega} \left( u \frac{\partial v}{\partial \vec{\nu}} - v \frac{\partial u}{\partial \vec{\nu}}\right) ds - \int \limits_{\Omega} u \Delta v \, dx. \label{2.3}
\end{equation}

Чтобы идти дальше и уверенно говорить о фундаментальном решении оператора Лапласа, сделаем краткий обзор основных положений теории обобщённых функций. Систематически обобщённые функции будут изучаться в четвёртой главе книги. Пусть $\Omega$ --- область в $\mathbb{R}^n$ (может быть всё пространство).

\begin{definition} \label{d.2.1}
Линейное пространство основных функций в области $\Omega$ состоит из всех финитных бесконечно дифференцируемых функций в этой области. Обозначение:  $C_0^{\infty}(\Omega)$ или $\mathcal{D}(\Omega)$.
\end{definition}

\begin{definition} \label{d.2.2}
Последовательность $\{ \varphi_{j} \}_{j \in \mathbb{N}} \subset \mathcal{D}(\Omega)$ сходится к основной функции $\varphi \in \mathcal{D}(\Omega)$, если выполнены два условия:
\begin{enumerate}
\item[(i)] существует такой компакт $K \subset \Omega$, что $\mathrm{supp}\, \varphi_j, \mathrm{supp}\, \varphi  \subset K$  для всех  $j \in \mathbb{N} $;
\item[(ii)] для любого мультииндекса $\alpha \in \mathbb{Z}^n_{+}$ имеет место равномерная сходимость $D^{\alpha}\varphi_j(x) \rightrightarrows D^{\alpha} \varphi(x)$, $x \in K$,  $j \to \infty$. 
\end{enumerate}
\end{definition}

\begin{definition} \label{d.2.3}
Линейный непрерывный функционал $f : \mathcal{D}(\Omega) \to \mathbb{R}$ называется обобщённой функцией или распределением. Непрерывность понимается в смысле Гейне: для любой последовательности \\$\left\{ \varphi_j \right\}_{j \in \mathbb{N}} \subset \mathcal{D}(\Omega)$, сходящейся к $\varphi \in \mathcal{D} (\Omega)$, $f \left( \varphi_{j} \right) \to f \left( \varphi \right) ,  j \to \infty$.
\end{definition}

При необходимости можно рассматривать обобщённые функции, принимающие комплексные значения (но соответствующий линейный функционал все равно остаётся линейным, но не антилиненйным); однако в настоявшей главе нам потребуются только вещественнозначные обобщённые функции. Примем традиционное обозначение $(f , \varphi) := f (\varphi)$. Также обозначим пространство всех обобщённых функций через $\mathcal{D}' (\Omega)$ и определим (если так можно сказать) $*$-слабую сходимость в этом пространстве.

\begin{definition} \label{d.2.4}
Последовательность обобщённых функций\\ $\left\{ f_j \right\}_{j \in \mathbb{N}} \subset \mathcal{D}'(\Omega)$ сходится к обобщённой функции $f \in \mathcal{D}'(\Omega)$, если для любой основной функции $\varphi \in \mathcal{D}(\Omega)$ имеем $(f_j , \varphi) \to (f , \varphi)$, $j \to \infty$.
\end{definition}

Все обобщённые функции можно разделить на регулярные и сингулярные. Первые получается из обычных (в смысле Ньютона или Лебега) функций и фактически отождествляются с ними, а вторые обычным функциям в таком смысле не соответствуют и не отождествляются с ними. Дадим точные определения.

\begin{definition} \label{d.2.5}
Измеримая функция $f: \Omega \to \mathbb{R}$ называется суммируемой в $\Omega$, если существует конечный интеграл $\int_{\Omega} |f(x)| dx$. Пространство суммируемых функций обозначается через $L_{1}(\Omega)$. Пространство локально суммируемых функций $L_{1,loc} (\Omega)$ состоит из всех измеримых функций $u :\Omega \to \mathbb{R}$, таких, что для любого открытого и ограниченного подмножества $Q \Subset \Omega$ имеем $\left. u \right|_{Q} \in L_1(Q)$.
\end{definition}

С каждой локально суммируемой функцией $u \in L_{1,loc} (\Omega)$ можно связать обобщённую функцию $U \in \mathcal{D}' \left( \mathbb{R} \right)$ по формуле: для любой  $\varphi \in \mathcal{D}(\Omega)$
\begin{equation}
 \left( U ,\varphi \right) : = \int \limits_{\Omega} u(x) \varphi(x) dx. \label{2.4}
\end{equation}
Стандартные теоремы функционального анализа утверждают, что формула \eqref{2.4} в самом деле задаёт обобщённую функцию (линейный непрерывных фунционал над всем пространством $\mathcal{D}(\Omega)$). И леммы Дюбуа-Реймона (см. пар. 2 в четвёртой главе) линейное соответствие $L_{1,loc} (\Omega) \to \mathcal{D}'(\Omega)$  инъективно, поэтому, на самом деле, локально суммируемые функции возможно отождествитесь с их представителями в пространстве $\mathcal{D}'(\Omega)$. Обобщённые функции, которые не получаются таким образом из локально суммируемых, называются сингулярными. Самая известная сингулярная обобщённая функция --- дельта-функция Дирака.

\begin{definition} \label{d.2.6}
Пусть $x^{0} \in \mathbb{R}^n$. Дельта-функцией Дирака (относительно точки $x^{0}$) называется обобщённая функция $\delta_{x^{0}} \in \mathcal{D}' \left( \mathbb{R}^n \right)$, действующая на основные функции по формуле $\left( \delta_{x^0}, \varphi \right):= \varphi \left( x^0 \right)$.
\end{definition}

Если $x^0$ --- начало координат, то пишут просто $\delta$. Сомнений в том, что $\delta$ (или $\delta_{x^{0}}$) --- обобщённая функция, нет, но её сингулярный характер требует некоторых объяснений. Предположим, что существует локально суммируемый представитель $d \in L_{1,loc} \left( \mathbb{R}^n \right)$. Тогда для любой $\varphi \in C^{\infty} \left( \mathbb{R}^n \right)   \int_{\mathbb{R}^n} d(x) \varphi(x) dx = \varphi(0)$ и для любой $\psi \in C^{\infty} \left( \mathbb{R}^n \right)$ верно, что $\int_{\mathbb{R}^n} d(x) |x|^2 \psi(x) dx = 0$, поскольку $\mathbb{R}^n \mapsto |x|^2 \psi(x)$ --- основная функция. По лемме Дюбуа-Реймона должно быть $d(x) |x|^2 = 0$ для почти всех $x \in \mathbb{R}^n$, откуда $d=0$ как элемент пространства $L_{1,loc} \left( \mathbb{R}^n \right)$, но дельта-функция не равна нулю.

Над обобщёнными функциями можно производить почти все те операции, что и над гладкими функциями. Остановимся на тех, которые будут нужны для определения фундаментального решения линейного дифференциального оператора, в частности, оператора Лапласа.

\begin{itemize}
\item Сложение и умножение на число (обобщённые функции --- линейные непрерывные функционалы).

\item Сужение $f \in \mathcal{D}' (\Omega)$ на открытое подмножество $Q \subset \Omega :  \left( \left. f \right|_{Q}, \varphi \right) := \left( f, \tilde{\varphi} \right)$ для любой $\varphi \in \mathcal{D}(Q)$, где $\tilde{\varphi}$ --- продолжение $\varphi$  нулём в $\Omega \backslash Q$.

\item Умножение на гладкую функцию $m \in C^{\infty}(\Omega) :  (mf , \varphi) := (f,m \varphi)$ для любой $\varphi \in \mathcal{D}(\Omega)$ (отметим, что произведение гладкой функции и основой функции --- функция основная).

\item Дифференцирование: для любого мультииндекса $\alpha \in \mathbb{Z}^{n}_{+}$ и любой функции $\varphi \in \mathcal{D} (\Omega)$
\begin{equation}
 \left( D^{\alpha} f , \varphi \right) := (-1)^{|\alpha|} \left( f, D^{\alpha} \varphi \right). \label{2.5}
\end{equation}
\end{itemize}

Кроме того, в третьей главе для объяснения метода конформных преобразований нам придётся делать замены переменных в обобщённых функциях. Пусть $\theta: Q \to \Omega$ --- $C^{\infty}$-диффеоморфизм областей $Q$ и $\Omega$, $f \in \mathcal{D}'(\Omega)$. Определим функцию $f \circ \theta \in \mathcal{D}'(Q)$, положив $\left( f \circ \theta, \varphi \right) := \left( f, \frac{1}{| \det \theta'|} \varphi \circ \theta^{-1} \right)$ для $\varphi \in \mathcal{D} (Q)$, где $\det \theta'$  --- якобиан отображения $\theta$.

Объясним определение производной и замены переменных. Если $f \in C^{\infty}(\Omega)$, $\alpha \in \mathbb{Z}^{n}_{+} $,  $|\alpha|=k$, то, интегрируя по частям $k$  раз, получим
\begin{equation*}
\int_{\Omega} (D^{\alpha} f) \varphi dx = (-1)^{|\alpha|} \int_{\Omega} f D^{\alpha} \varphi dx
\end{equation*}
для любой основной функции $\varphi \in \mathcal{D}(\Omega)$  (благодаря финитности $\varphi$, граничные интегралы исчезают). Таким образом, для гладких функций формула \eqref{2.5} --- следствие из формулы интегрирования по частям, поэтому естественно положить её в основу определения производных обобщённых функций.

Если $f \in L_{1,loc} (\Omega)$, $\theta: Q \ni x \mapsto y \in \Omega$ --- диффеоморфизм, то замена переменных в интеграле Лебега даёт для любой функции $\varphi \in \mathcal{D}(Q)$
\begin{equation*}
 \int \limits_{Q} f \left( \theta(x) \right) \varphi (x) dx = \int \limits_{\Omega} f(y) \varphi \left( \theta^{-1}(y) \right) \frac{1}{|\det \theta'|} dy. 
\end{equation*}
Таким образом, для регулярных обобщённых функций указанная ранее формулы замены переменных --- следствие хорошо известного свойства интеграла.

Рассмотрим линейный дифференциальный оператор $\mathcal{L}$, действующий на обобщённые функции $u \in \mathcal{D}'(\Omega)$ по формуле $\mathcal{L} u = \sum \limits_{|\alpha| \leqslant m} a_{\alpha}(x) D^{\alpha} u$, где $a_{\alpha} \in C^{\infty}(\Omega)$. Оператор $\mathcal{L}^{*}: \mathcal{D}'(\Omega) \to \mathcal{D}'(\Omega)$ формально сопряжён к $\mathcal{L}$, если для всех $u \in \mathcal{D}'(\Omega) ,  \varphi \in \mathcal{D} (\Omega)$ имеем $\left( \mathcal{L} u, \varphi \right) = \left( u,\mathcal{L}^{*} \varphi \right)$. Из определения умножения на гладкую функции и дифференцирования обобщённой функции следует, что формально сопряжённый к ЛДО --- тоже ЛДО, и он выражается формулой
\begin{equation*}
\mathcal{L} u = \sum \limits_{|\alpha| \leqslant m} (-1)^{|\alpha|} D^{\alpha} \left( a_{\alpha}(x) u \right).
\end{equation*}
Например, если $\mathcal{L} u = -\mathrm{div} \left( A(x) \nabla u \right) + \vec{b}(x) \cdot \nabla u + c(x) u$ --- ЛДО второго порядка, то сопряжённый к нему имеет вид
\begin{equation*}
 \mathcal{L}^{*} u = -\mathrm{div} \left( A(x) \nabla u \right) - \mathrm{div} \left( \vec{b}(x) \cdot \nabla u \right) + \left( c(x) + \mathrm{div} \, \vec{b}(x) \right) u. 
\end{equation*}
Очевидно, что $\mathcal{L}^{**}=\mathcal{L}$. Оператор формально самосопряжён, если $\mathcal{L}^{*} = \mathcal{L}$. Указанный выше ЛДО второго порядка формально самосопряжён тогда и только тогда, когда $\vec{b}(x) \equiv 0$. В частности, оператор Лапласа формально самосопряжён.

\begin{definition} \label{d.2.7}
 Фундаментальным решением ЛДО $\mathcal{L}$  с гладкими коэффициентами в области $\Omega \subset \mathbb{R}^n$ называется семейство обобщённых функций  $E \left( \cdot, x^0 \right) \in \mathcal{D}'(\Omega),  x^0 \in \Omega$, с тем свойством, что $\mathcal{L} E \left( \cdot, x^0 \right) = \delta_{x^0}$ для любой $x^0 \in \Omega$. Или, другими словами, для любых $x^0 \in \Omega$  и  $\varphi \in \mathcal{D}(\Omega)$ 
\begin{equation*}
 \left( \mathcal{L} E \left( \cdot, x^0 \right), \varphi \right) = \left( E \left( \cdot,x^0 \right), \mathcal{L}^{*} \varphi \right) = \varphi \left( x^0 \right).
 \end{equation*}
\end{definition}

Однако, если $\Omega = \mathbb{R}^n$  и оператор $\mathcal{L}$ имеет постоянные коэффициенты, то семейство можно заменить на одну обобщённому функцию $\tilde{E} \in \mathcal{D}' \left( \mathbb{R}^n \right)$ и получить его по формуле $E \left( x,x^0 \right) =\tilde{E} \left( x-x^0 \right)$. В четвёртой главе будет доказано, что всякий ЛДО с постоянными коэффициентами имеет фундаментальное решение. Разумеется, фундаментальное решение, вообще говоря, не единственно (оно определено с точностью до ядра оператора, т.е. решений уравнения $\mathcal{L} u = 0$). Построение фундаментального решения может быть сложной и даже очень сложной задачей.

Естественно предположить, что существует фундаментальное решение оператора Лапласа $E=E(x)$ с радиальной симметрией. Поскольку $\Delta E =0$  при $x \ne 0$, найдём все радиально-симметричные решения уравнения Лапласа в проколотом пространстве $\mathbb{R}^n \backslash \{ 0 \}$. Обозначим $r = \sqrt{x_1^2 + \ldots + x_n^2}$,  $u(x) = v(r)$, и получим выражение для оператора Лапласа. Имеем
\begin{equation*}
 \frac{\partial u}{\partial x_j} = \frac{d v}{d r} \frac{x_j}{r },\, \frac{\partial^2 u}{\partial x_j^2} = \frac{\partial}{\partial x_j} \left( \frac{d v}{d r} \frac{x_j}{r } \right) = \frac{d^2 v}{d r^2} \frac{x_j^2}{r^2} + \frac{d v}{d r} - \frac{d v}{d r} \frac{x_j^2}{r^3},
 \end{equation*}
откуда $\Delta u= \sum \limits_{j=1}^{n} \frac{\partial^2 u}{\partial x_j^2} = \frac{d^2 v}{dr^2} + \frac{n-1}{r} \frac{dv}{dr}$. Функция $v$ удовлетворяет дифференциальному уравнению вида $\frac{d^2 v}{dr^2} + \frac{n-1}{r} \frac{dv}{dr} = 0$, его общее решение: $v=C_1 \ln r + C_2$  при  $n=2$, $v= C_1 r^{2-n} + C_2$ при $n>2$, где $C_1$, $C_2$  --- произвольные независимые постоянным. Если существует радиально-симметричное фундаментальное решение оператора Лапласа, то оно имеет вид $E(x) = C_1(2) \ln |x| + C_2$ ($n=2$) или $E(x) = C_1(n) |x|^{2-n} + C_2$ ($n>2$). Константу $C_2$ можно положить равной нулю, но константа $C_1=C_1(n)$ должна быть определена из уравнения для фундаментального решения.

\begin{proposition}\label{p.2.1}
Функция
\begin{equation}
E(x) = \left\{ \begin{aligned} & -\frac{1}{2\pi}\ln \frac{1}{|x|},\, n=2,\\ & - \frac{1}{(n-2) \omega_{n-1}} \frac{1}{|x|^{n-2}},\, n>2,\\ \end{aligned} \right. \label{2.6}
\end{equation}
где $\omega_{n-1}$ --- гиперплощадь ($(n-1)$-мерный объём) гиперсферы $\mathbb{S}^{n-1} = \left\{ x \in \mathbb{R}^n: |x|=1\right\}$, есть фундаментальное решение оператора Лапласа в евклидовом пространстве $\mathbb{R}^n$.
\end{proposition}

Вычислим $\omega_{n-1}$. Интеграл $\int_{-\infty}^{\infty} e^{-x^2} dx = \sqrt{\pi}$ даёт
\begin{equation*}
 \pi^{\frac{n}{2}} =  \int_{-\infty}^{\infty} \ldots \int_{-\infty}^{\infty} e^{-x_1^2-\ldots-x_n^2} dx_1 \ldots dx_n= \omega_{n-1} \int_{0}^{\infty} r^{n-1} e^{-r^2} dr = \frac{1}{2}\omega_{n-1} \Gamma (n/2),
\end{equation*}
где $\Gamma(z) = \int_{0}^{\infty} t^{z-1} e^{-t} dt$ --- гамма-функция Эйлера. Отсюда $\omega_{n-1}= \frac{2 \pi^{\frac{n}{2}}}{\Gamma(n/2)}$; имеем $\omega_1=2 \pi$, $\omega_2 = 4 \pi$, $\omega_3=2 \pi^2$  и т.д.

Перед доказательством утверждения \ref{p.2.1} отметим некоторые свойства функции \eqref{2.6}, легко следующие из явной формулы для неё:
\begin{itemize}
\item $E(x) = \mathcal{E} (|x|)$, где $\mathcal{E}(r) = -\frac{1}{2\pi} \ln \frac{1}{r}$, если $n=2$, $\mathcal{E}(r) = - \frac{1}{(n-2) \omega_{n-1}} \frac{1}{r^{n-2}}$ при $n>2$;

\item $\Delta E = 0$  в  $\mathbb{R}^n \backslash \{ 0 \}$;

\item $E \in C^{\infty} \left( \mathbb{R}^n \backslash {0} \right) \cap L_{1,loc} \left( \mathbb{R}^n \right)$.
\end{itemize}

\begin{proof} Из определения фундаментального решения и формальной самосопряжённости оператора Лапласа явствует: нужно доказать, что для любой $\varphi \in \mathcal{D} \left( \mathbb{R}^n \right)$  $\int_{\mathbb{R}^n} E \Delta \varphi\, dx = \varphi(0)$. Локальная суммируемость функции $E$ позволяет представить последний интеграл в виде предела
\begin{equation*}
 \int \limits_{\mathbb{R}^n} E \Delta \varphi\, dx = \lim \limits_{\varepsilon \to 0} \int \limits_{\mathbb{R}^n \backslash B_{\varepsilon}} E \Delta \varphi\, dx, 
\end{equation*}
где $B_{\varepsilon} = \left\{ x \in \mathbb{R}^n: |x| <\varepsilon \right\}$ --- шар с центром в начале и радиусом $\varepsilon$. Применим вторую формулу Грина, учитывая финитность функции $\varphi$:
\begin{equation}
\int \limits_{\mathbb{R}^n \backslash B_{\varepsilon}} E \Delta \varphi \, dx = \int \limits_{\mathbb{R}^n \backslash B_{\varepsilon}} \varphi \Delta E \, dx + \int \limits_{S_{\varepsilon}} \left( E \frac{\partial \varphi}{\partial \vec{\nu}'} - \varphi \frac{\partial E}{\partial \vec{\nu}'} \right) ds. \label{2.7}
\end{equation}
Здесь $S_{\varepsilon} = \partial B_{\varepsilon}= \left\{ x \in \mathbb{R}^n: |x| =\varepsilon \right\}$ --- сфера с центром в начале координат и радиусом $\varepsilon$, $\vec{\nu}'$  --- единичная внутренняя (по отношению к шару) нормаль к $S_{\varepsilon}$.

Интеграл по $\mathbb{R}^n \backslash B_{\varepsilon}$ в правой части \eqref{2.7} равен нулю, потому что $E$ --- гармоническая функция в $\mathbb{R}^n \backslash \{ 0\}$. Следовательно,
\begin{equation}
\int \limits_{\mathbb{R}^n \backslash B_{\varepsilon}} E \Delta \varphi dx = \int \limits_{S_{\varepsilon}} E \frac{\partial \varphi}{\partial \vec{\nu}'}\, ds-\int \limits_{S_{\varepsilon}} \varphi \frac{\partial E}{\partial \vec{\nu}'} \, ds. \label{2.8} 
\end{equation}
Оценим первый интеграл в правой части равенства \eqref{2.8}:
\begin{equation*}
\left| \int \limits_{S_{\varepsilon}} E \frac{\partial \varphi}{\partial \vec{\nu}'}\, ds \right| \leqslant | \mathcal{E}(\varepsilon) | \int \limits_{S_{\varepsilon}} \left| \frac{\partial \varphi}{\partial \vec{\nu}'}\right| ds \leqslant |\mathcal{E}(\varepsilon)| \omega_{n-1} \varepsilon^{n-1}\sup \limits_{S_{\varepsilon}} \left| \frac{\partial \varphi}{\partial \vec{\nu}'}\right| = C_{\varphi} \varepsilon^{n-1} |\mathcal{E}(\varepsilon)|.
\end{equation*}
Мы видим, что $\int_{S_{\varepsilon}} E \frac{\partial \varphi}{\partial \vec{\nu}'}\, ds \to 0$  при $\varepsilon \to 0$ (константа $C_{\varphi}$ зависит от $\varphi$, но не от  $\varepsilon$).

Рассмотрим предел от второго интеграла в правой части равенства (2.8):
\begin{equation*}
\lim \limits_{\varepsilon \to 0} \left(-\int \limits_{S_{\varepsilon}} \varphi \frac{\partial E}{\partial \vec{\nu}'} \, ds \right)= \lim \limits_{\varepsilon \to 0} \frac{\varepsilon^{n-1}}{\omega_{n-1}} \int \limits_{S_{\varepsilon}} \varphi \, ds= \lim \limits_{\varepsilon \to 0} \frac{\varepsilon^{n-1}}{\omega_{n-1}} \varphi\left(\xi_{\varepsilon}\right) \omega_{n-1}\varepsilon^{n-1}, 
\end{equation*}
где $\xi_{\varepsilon} \in S_{\varepsilon}$, и мы применили теорему о среднем для интеграла. Но $\xi_{\varepsilon} \to 0$, если $\varepsilon \to 0$, поэтому последний предел равен $\varphi(0)$. Итак,
\begin{equation*}
 \int \limits_{\mathbb{R}^n} E(x) \Delta \varphi\, dx = \lim \limits_{\varepsilon \to 0} \int \limits_{\mathbb{R}^n \backslash B_{\varepsilon}} E(x) \Delta \varphi\, dx =\varphi(0) 
\end{equation*}
и утверждение \ref{p.2.1} доказано.
\end{proof}

\section{Интегральное представление функций. Потенциалы}
\markright{\thesection\ Интегральное представление. Потенциалы}

Фундаментальное решение оператора Лапласа, построенное в предыдущем параграфе, даёт возможность получить интегральное представление функций в заданной области $\Omega \Subset \mathbb{R}^n$. Под интегральными представлениями понимаются формулы, выражающие функции через интегралы, которые содержат оператор в области и операторы на границе от данной функции.

Итак, пусть $\Omega \Subset \mathbb{R}^n$, $\partial \Omega \in C^{1}$, $u \in C^{2} \left( \overline{\Omega} \right)$, $B_{\varepsilon} \left( x^0 \right) \Subset \Omega$. Через  $\Omega_{\varepsilon} = \Omega \backslash \overline{B}_{\varepsilon} \left( x^0 \right)$ обозначим $\varepsilon$-перфорацию области $\Omega$ в точке $x^0$. Имеем $\partial \Omega_{\varepsilon} = \partial \Omega \sqcup S_{\varepsilon} \left( x^0 \right)$ (рис. 2.1). Применим вторую формулу Грина \eqref{2.3} к функциям $u$  и $v(x)=E \left( x-x^0 \right)$ в $\Omega_{\varepsilon}$, отметив сначала, что для любых $x^0 \in \mathbb{R}^n$  и $\varphi \in \mathcal{D} \left( \mathbb{R}^n \right)$  $\int_{\mathbb{R}^n} E\left( x-x^0 \right) \varphi(x)\, dx = \varphi \left( x^0\right)$. Имеем  $0=\int_{\Omega_{\varepsilon}} u(x) \Delta E\left( x-x^0 \right) \, dx =$
\begin{equation}
 = \int \limits_{\partial \Omega} \left( u \frac{\partial E}{\partial \vec{\nu}} - E \frac{\partial u}{\partial \vec{\nu}}\right) ds +\int \limits_{S_{\varepsilon}\left( x^0 \right)} \left( u \frac{\partial E}{\partial \vec{\nu}'} - E \frac{\partial u}{\partial \vec{\nu}'}\right) ds + \int \limits_{\Omega_{\varepsilon}} E \left( x-x^0 \right) \Delta u(x) \, dx, \label{2.9}
\end{equation}
где $\vec{\nu}'$ --- внутренняя нормаль к $S_{\varepsilon} \left( x^0 \right)$ (и внешняя для области $\Omega_{\varepsilon}$). Рассуждая как при доказательстве утверждения \ref{p.2.1}, можно убедиться, что при $\varepsilon \to 0$ интеграл по $S_{\varepsilon} \left( x^0 \right)$ стремится к $-u \left(x^0 \right)$. Тогда сделаем предельный переход в \eqref{2.9}, устремив $\varepsilon$ к нулю, и получим формулу интегрального представления функций.

\begin{proposition}\label{p.2.2}
Пусть $\Omega \Subset \mathbb{R}^n$, $\partial \Omega \in C^{1}$, тогда для любой функции $u \in C^{2} \left( \overline{\Omega} \right)$ в любой точке $x^0 \in \Omega$ верна формула интегрального представления
\begin{equation}
 u \left( x^0 \right) =  \int \limits_{\Omega} E \left( x-x^0 \right) \Delta u(x)\, dx + \int \limits_{\partial \Omega}\frac{\partial E}{\partial \vec{\nu}} u(x)  ds - \int \limits_{\partial \Omega} E \left( x-x^0 \right) \frac{\partial u}{\partial \vec{\nu}} ds.\label{2.10}
 \end{equation}
 \end{proposition}
 
В частности, если $u \in C^{2} \left( \overline{\Omega} \right)$ --- гармоническая функция в $\Omega$, $\left. u \right|_{\partial \Omega}= \varphi$, $\left. -\frac{\partial u}{\partial \vec{\nu}} \right|_{\partial \Omega}= \psi$, то формула интегрального представления гармонической функции имеет вид
\begin{equation}
 u \left( x^0 \right) = \int \limits_{\partial \Omega} \varphi(x) \frac{\partial E}{\partial \vec{\nu}} ds + \int \limits_{\partial \Omega}E \left( x- x^0 \right)  \psi(x)  ds. \label{2.11}
 \end{equation}
 На первый взгляд формула \eqref{2.11} даёт общее решение уравнения Лапласа в области $\Omega$, а именно, описывает все дважды непрерывно дифференцируемые в $\overline{\Omega}$ гармонические функции. На самом деле это не так: функции $\varphi$ и $\psi$ в формуле \eqref{2.11} зависят друг от друга. Например, если задать функцию $\varphi$, т.е. краевое условие Дирихле $\left. u \right|_{\partial \Omega}= \varphi$, то, как будет показано в следующих параграфах этой главы, соответствующая краевая задача будет иметь единственное решение. Следовательно, вторая функция $\psi$ однозначно зависит от $\varphi$. Если уж записывать общее решение уравнения Лапласа, то делать это нужно с помощью функции Грина задачи Дирихле, либо задачи Неймана, что и будет сделано в третьей главе.

Однако, это не обесценивает формулу интегрального представления функций. Интегралы, из которых она состоит, в размерности $n=3$  физики называют потенциалами Ньютона--Кулона, а в размерности $n=2$ --- логарифмическими потенциалами. Дадим соответствующие определения, а также приведём без доказательств свойства потенциалов.

{\bf Объёмный потенциал.} Пусть $\Omega \Subset \mathbb{R}^3$, $\partial \Omega \in C^1$, $\rho_0 \in C \left( \overline{\Omega} \right)$, $Q = \mathbb{R}^3 \backslash \overline{\Omega}$. Тогда объёмный потенциал $V_0 \left( x^0 \right) = \iiint_{\Omega} \frac{\rho_0(x)}{ \left| x-x^0 \right|} dx$  существует для всех точек $x^0 \in \mathbb{R}^3$  и имеет свойства:
\begin{enumerate}
\item[1)] $V_0 \in C^1 \left( \mathbb{R}^3\right) \cap C_{b} \left( \mathbb{R}^3 \right)$  (непрерывная дифференцируемость и ограниченность в пространстве);
\item[2)] $V_0 \left( x^0 \right) = O \left( \frac{1}{|x^0|}\right)$, если $|x^0| \to \infty$;
\item[3)] $\Delta V_0=0$  в  $Q$;
\item[4)] пусть $\rho_0 \in C \left( \overline{\Omega} \right) \cap C^{1} (\Omega)$, тогда $V_0 \in C^{2}(\Omega)$ и удовлетворяет в $\Omega$ уравнению Пуассона $\Delta V_0 = - 4 \pi \rho_0(x)$.
\end{enumerate}

\begin{example}\label{e.2.1}
Пусть $B_{R} = \left\{ x \in \mathbb{R}^3: |x| < R \right\}$ --- шар радиуса $R$ в трёхмерном евклидовом пространстве $\rho_0(x) = \rho_0 = \mathrm{const}$. Найти объёмный потенциал с плотностью $\rho_0$  для шара $B_R$.

{\it Решение.} Имеем $V_0 \left( x^0 \right) = \iiint_{B_R} \frac{\rho_0}{ \left| x-x^0 \right|} dx_1 dx_2 dx_3$, но найти интеграл непосредственно довольно сложно. Заметим, во-первых, что  потенциал есть радиально-симметричная функция, т.е. его можно представить в виде $V_0 \left( x^0 \right) = v(r)$, где $r = |x^0|$ --- сферический радиус. Действительно, пусть $S$ --- матрица, задающая ортогональное преобразование в $\mathbb{R}^n$, тогда, осуществляя замену $x=S x'$  в интеграле, получим $V_0 \left( Sx^0 \right) =$
\begin{equation*}
= \int_{B_R} \frac{\rho_0 dx_1 dx_2 dx_3}{ \left| x-S x^0 \right|} =  \int_{B_R} \frac{\rho_0 | \mathrm{det}\, S| dx'_1 dx'_2 dx'_3}{ \left| Sx' - S x^0 \right|} =  \int_{B_R} \frac{\rho_0  dx'_1 dx'_2 dx'_3}{ \left| x' - x^0 \right|}  = V_0 \left( x^0 \right),
\end{equation*}
откуда следует, что функция $V_0$ принимает постоянное значение на каждой сфере с центром в начале координат, т.е. $V_0$ имеет радиальную симметрию.

Во-вторых, из третьего и четвёртого свойств объёмного потенциала следует, что $\Delta V_0 = 0$ при $r>R$ и $\Delta V_0 = -4 \pi \rho_0$ для $0<r<R$. Отсюда
\begin{equation*}
v(r) = \left\{
\begin{aligned}
& \frac{A}{r} +B,\, r>R,\\
& \frac{C}{r} + D - \frac{2}{3} \pi \rho_0 r^2,\, 0<r<R,\\
\end{aligned}
\right.
\end{equation*}
где $A$, $B$, $C$ и $D$ --- четыре неопределённых параметра. Чтобы их найти, обратимся к другим свойствам объёмного потенциала.

Из второго свойства следует, что $B=0$, иначе мы не получим правильную асимптотику потенциала на бесконечности. Из первого свойства следует, что функция $V_0$ должна быть непрерывно дифференцируемой на граничной сфере $S_R = \partial B_{R}$. Имеем систему из двух уравнений
\begin{equation}
\left\{
\begin{aligned}
& \frac{A}{R} = \frac{C}{R} + D - \frac{2}{3} \pi \rho_0 R^2,\\
& -\frac{A}{R^2} = -\frac{C}{R^2} - \frac{4}{3} \pi \rho_0 R.\\
\end{aligned}
\right.
\label{2.12}
\end{equation}
Прежде чем решать систему \eqref{2.12} заметим ещё, что функция $V_0$ не должна иметь особенностей при $r \to 0$, поэтому $C=0$. Но тогда $A = \frac{4}{3} \pi \rho_0 R^3$, $D = 2 \pi \rho_0 R^2$. Итак,
 \begin{equation*}
V_0 \left( x^0 \right)= \left\{
\begin{aligned}
& \frac{4 \pi \rho R^3}{3 | x^0 |},\, |x^0|  \geqslant R,\\
& 2 \pi \rho_0 R^2 - \frac{2}{3} \pi \rho_0 |x^0|^2,\, 0 \leqslant |x^0|  \leqslant R.\\
\end{aligned}
\right.
\end{equation*}
\end{example}

{\bf Потенциал простого слоя.} Пусть $\Omega \Subset \mathbb{R}^3$, $\partial \Omega \in C^1$, $\rho_1 \in C (\partial \Omega)$, $Q = \mathbb{R}^3 \backslash \partial \Omega$. Тогда потенциал простого слоя $V_1 \left( x^0 \right) = \iint_{\partial \Omega} \frac{\rho_1(x)}{ \left| x-x^0 \right|} ds$  существует для всех точек $x^0 \in \mathbb{R}^3$. Отметим его свойства:
\begin{enumerate}
\item[1)] $V_1 \in C_{b} \left( \mathbb{R}^3 \right)$  (непрерывность  и ограниченность в пространстве);
\item[2)] $V_1 \left( x^0 \right) = O \left( \frac{1}{|x^0|}\right)$, если $|x^0| \to \infty$;
\item[3)] $\Delta V_1=0$  в  $Q$;
\item[4)] пусть $x^0 \in \partial \Omega$, тогда существует нормальная производная извне (из внешней для $\partial \Omega$ области) $\partial_{\vec{\nu}} V_{1}^{+} \left(x^0 \right)$, изнутри (относительно $\Omega$) $\partial_{\vec{\nu}} V_{1}^{-} \left(x^0 \right)$ и они связаны соотношением $\partial_{\vec{\nu}} V_{1}^{+} \left(x^0 \right) - \partial_{\vec{\nu}} V_{1}^{-} \left(x^0 \right) = -4 \pi \rho_1 \left( x^0 \right)$ (разрыв нормальной производной потенциала простого слоя).
\end{enumerate}

{\bf Потенциал двойного слоя.} Пусть $\Omega \Subset \mathbb{R}^3$, $\partial \Omega \in C^1$, $\rho_2 \in C (\partial \Omega)$, $Q = \mathbb{R}^3 \backslash \partial \Omega$. Тогда потенциал двойного слоя $V_2 \left( x^0 \right) = \iint_{\partial \Omega} \rho_2(x) \partial_{\vec{\nu}}  \left| x-x^0 \right|^{-1} ds$  существует для всех точек $x^0 \in \mathbb{R}^3$. Свойства потенциала двойного слоя:
\begin{enumerate}
\item[1)] $V_2 \in L_{\infty} \left( \mathbb{R}^3 \right)$  (ограниченность в пространстве);
\item[2)] $V_2 \left( x^0 \right) = O \left( \frac{1}{|x^0|^2}\right)$, если $|x^0| \to \infty$\footnote{Обратите внимание на отличие асимптотики на бесконечности потенциала двойного слоя от соответствующей асимптотики других потенциалов Ньютона--Кулона.};
\item[3)] $\Delta V_2=0$  в  $Q$;
\item[4)] пусть $x^0 \in \partial \Omega$, $\rho_2 \in C^1 (S)$, тогда потенциал двойного слоя имеет предел извне $ V_{2}^{+} \left(x^0 \right)$, предел изнутри $ V_{2}^{-} \left(x^0 \right)$, которые связаны формулами
\begin{equation}
\label{2.13}
\left\{
\begin{aligned}
& V_{2}^{+} \left(x^0 \right)  = 2 \pi \rho_2 \left( x^0 \right)+ V_{2} \left(x^0 \right) ,\\
& V_{2}^{-} \left(x^0 \right)  = -2 \pi \rho_2 \left( x^0 \right)+ V_{2} \left(x^0 \right).\\
\end{aligned}
\right.
\end{equation}
\end{enumerate}

Четвёртое свойство потенциала двойного слоя гласит, что он терпит разрыв первого рода на замкнутой поверхности $\partial \Omega$. Из \eqref{2.13} следует, что $V_{2} \left( x^0 \right) = \frac{V_{2}^{+} \left(x^0 \right)+ V_{2}^{-} \left(x^0 \right)}{2}$. Все потенциалы Ньютона--Кулона имеют электростатическую интерпретацию: например, если электрические заряды распределены на $\partial \Omega$, то создаваемое ими поле имеет потенциал простого слоя $V_1$.

{\bf Логарифмический потенциал площади.} Пусть $\Omega \Subset \mathbb{R}^2$, $\partial \Omega \in C^1$, $\rho_0 \in C \left( \overline{\Omega} \right)$, $Q = \mathbb{R}^2 \backslash \overline{\Omega}$. Тогда логарифмический потенциал площади $V_0 \left( x^0 \right) = \iint_{\Omega} \rho_0(x) \ln \left| x-x^0 \right|^{-1} dx$  существует для всех точек $x^0 \in \mathbb{R}^2$  и имеет свойства:
\begin{enumerate}
\item[1)] $V_0 \in C^1 \left( \mathbb{R}^2\right)$  (непрерывная дифференцируемость на плоскости);
\item[2)] $V_0 \left( x^0 \right) = O \left( \ln \frac{1}{|x^0|}\right)$, если $|x^0| \to \infty$;
\item[3)] $\Delta V_0=0$  в  $Q$;
\item[4)] пусть $\rho_0 \in C \left( \overline{\Omega} \right) \cap C^{1} (\Omega)$, тогда $V_0 \in C^{2}(\Omega)$ и удовлетворяет в $\Omega$ уравнению Пуассона $\Delta V_0 = - 2 \pi \rho_0(x)$.
\end{enumerate}

{\bf Логарифмический потенциал простой кривой.} Пусть $\Omega \Subset \mathbb{R}^2$, $\Gamma=\partial \Omega \in C^1$ --- дизъюнктное объединение конечного числа замкнутых кривых Жордана, $\rho_1 \in C (\Gamma)$, $Q = \mathbb{R}^2 \backslash \overline{\Omega}$. Тогда логарифмический потенциал простой кривой $V_1 \left( x^0 \right) = \oint_{\Gamma} \rho_1(x) \ln \left| x-x^0 \right|^{-1} dx$  существует для всех точек $x^0 \in \mathbb{R}^2$. Отметим свойства потенциала простой кривой:
\begin{enumerate}
\item[1)] $V_1 \in C \left( \mathbb{R}^2\right)$  (непрерывность на плоскости);
\item[2)] $V_1 \left( x^0 \right) = O \left( \ln \frac{1}{ |x^0|}\right)$, если $|x^0| \to \infty$;
\item[3)] $\Delta V_1=0$  в  $Q$;
\item[4)] разрыв нормальной производной: пусть $x^0 \in \Gamma$, тогда потенциал простой кривой имеет нормальные производные (по направлению внешней нормали) извне $\partial_{\vec{\nu}} V_{1}^{+} \left(x^0 \right)$, изнутри $\partial_{\vec{\nu}} V_{1}^{-} \left(x^0 \right)$, связанные соотношением $\partial_{\vec{\nu}} V_{1}^{+} \left(x^0 \right) - \partial_{\vec{\nu}} V_{1}^{-} \left(x^0 \right) = -2 \pi \rho_1 \left( x^0 \right)$.
\end{enumerate}

{\bf Логарифмический потенциал двойной кривой.} Пусть $\Omega \Subset \mathbb{R}^2$, $\Gamma=\partial \Omega \in C^1$ --- дизъюнктное объединение конечного числа замкнутых кривых Жордана, $\rho_2 \in C (\Gamma)$, $Q = \mathbb{R}^2 \backslash \overline{\Omega}$. Тогда логарифмический потенциал двойной кривой $V_2 \left( x^0 \right) =V_2 \left( x^0 \right) = \oint_{\Omega} \rho_2(x) \partial_{\vec{\nu}} \ln \left| x-x^0 \right|^{-1} ds $  существует для всех точек $x^0 \in \mathbb{R}^2$ и имеет следующие свойства:
\begin{enumerate}
\item[1)] $V_2 \in L_{\infty} \left( \mathbb{R}^2\right)$  (ограниченность на плоскости);
\item[2)] $V_2 \left( x^0 \right) =  O \left( \frac{1}{|x^0|}\right)$, если $|x^0| \to \infty$\footnote{Как и потенциал двойного слоя, потенциал двойной кривой имеет отличную от других потенциалов асиптотику на бесконечности, кроме того, потенциалы площади и простой кривой, вообще говоря, не ограничены на плоскости.};
\item[3)] $\Delta V_2=0$  в  $Q$;
\item[4)] разрыв потенциала двойной кривой: пусть $x^0 \in \Gamma$, тогда потенциал двойной кривой имеет предел извне $V_{2}^{+} \left(x^0 \right)$, изнутри $V_{2}^{-} \left(x^0 \right)$, связанные формулами Сохоцкого
\begin{equation}
\label{2.14}
\left\{
\begin{aligned}
& V_{2}^{+} \left(x^0 \right)  = \pi \rho_2 \left( x^0 \right)+ V_{2} \left(x^0 \right) ,\\
& V_{2}^{-} \left(x^0 \right)  = -\pi \rho_2 \left( x^0 \right)+ V_{2} \left(x^0 \right).\\
\end{aligned}
\right.
\end{equation}
\end{enumerate}
Таким образом, логарифмический потенциал двойной кривой имеет разрыв второго рода на границе области, а из формул Сохоцкого \eqref{2.14} следует, что его значение в граничных точках равно среднему арифметическому внутреннего и внешнего пределов: $V_{2} \left( x^0 \right) = \frac{V_{2}^{+} \left(x^0 \right) + V_{2}^{-} \left(x^0 \right)}{2}$.

Если искать потенциал двойной кривой по формуле \eqref{2.15}, то необходимо последовательно рассмотреть три случая: а) $ \left|x^0 \right|<R$ (точка находится внутри окружности); б) $\left| x^0 \right|=R$ (точка лежит на окружности); в) $\left| x^0 \right| > R$ (точка находится вне окружности). Альтернативное решение использует свойства потенциала двойной кривой. Однако, в этом примере функция $V_2$ не имеет радиальной симметрии, поэтому нужна дополнительная идея о структуре потенциала двойной кривой.

Подсказку даёт оператор Лапласа в полярных координатах
\begin{equation}
\label{2.16} \Delta V_2 = \frac{\partial^2 V_2}{\partial r^2} + \frac{1}{r} \frac{\partial V_2}{\partial r} + \frac{1}{r^2} \frac{\partial^2 V_2}{\partial \theta^2}.
\end{equation}
Заметим, что здесь нет смешанной производной $\frac{\partial^2 V_2}{\partial r \partial \theta}$, и причина в ортогональности полярной системы координат (формула \eqref{2.16} будет получена в пар. 7 шестой главы при обсуждении оператора Лапласа--Бельтрами на римановых многообразиях). Будем искать потенциал двойной кривой в виде $V_{2}(r,\theta) = Z(r) Y(\theta)$, где $Z$, $Y$ --- неизвестные функции, и воспользуемся третьим свойством, согласно которому $\Delta V_2=0$, если $r \ne R$. Подстановка в уравнение Лапласа даёт
\begin{equation}
\label{2.17}
Z''(r) Y(\theta) + \frac{1}{r} Z'(r) Y(\theta) + \frac{1}{r^2} Z(r) Y''(\theta) = 0.
\end{equation}
Разделим переменные, т.е. преобразуем уравнение \eqref{2.17} таким образом, чтобы его левая часть зависела только от переменной $r$, а правая часть --- только от $\theta$:
\begin{equation}
\label{2.18}
\frac{r^2 Z''(r) + rZ'(r)}{Z(r)}= -\frac{Y''(\theta}{Y(\theta)}.
\end{equation}
Так как $r$, $\theta$ --- независимые переменные, уравнение \eqref{2.18} имеет решение только в том случае, когда обе его части равны не зависят от $r$ и $\theta$, т.е. равны константе (обозначим её $\lambda$):
\begin{equation}
\label{2.19}
\frac{r^2 Z''(r) + rZ'(r)}{Z(r)}= -\frac{Y''(\theta}{Y(\theta)} = \mathrm{const} = \lambda.
\end{equation}
Из \eqref{2.19} следует уравнение $Y'' + \lambda Y = 0$, и легко видеть, что при $\lambda=1$ оно имеет частное решение, совпадающее с плотностью: $Y(\theta) = \rho_2(\theta) = \cos \theta$. Общее решение уравнения $r^2 Z'' + r Z' -Z = 0$ имеет вид $Z(r) = \frac{C_1}{r} + C_2$. \section{Принцип максимума и его следствия}
\markright{\thesection\ Принцип максимума и его следствия}

Принцип максимума для гармонических функций доказан К.Ф. Гауссом в 1840-м году в []. Следуя классику, мы выведем его из теоремы о среднем по шару, но не только для гармонических функций: для наших целей наряду с уравнением Лапласа $\Delta u =0$ нужно рассматривать неравенства $\Delta u \geqslant 0$ и $\Delta u \leqslant 0$. Дадим соответствующее определение.

\begin{definition} \label{d.2.8}
Решение уравнения Лапласа $\Delta u =0$ называется гармонической функцией, решение неравенства $\Delta u \geqslant 0$ --- субгармонической функцией, решение неравенства $\Delta u \leqslant 0$ --- супергармонической функцией.
\end{definition}

Сосредоточимся на субгармонических функциях (соответствующие результаты о супергармонических и гармонических функциях будут простыми следствиями утверждений о первых). Заметим сначала один факт: если $u \in C^{2} \left( \overline{\Omega} \right)$ ---  субгармоническая в области $\Omega \Subset \mathbb{R}^n$ с границей класса $C^{1}$, то из второй формулы Грина вытекает $\int_{\partial \Omega} \frac{\partial u}{\partial \vec{\nu}} ds \geqslant 0$.

\begin{theorem}\label{t.2.1}
Пусть $B_{R}  = \left\{ x \in \mathbb{R}^n: |x|<R\right\}$, $S_{R} = \partial B_{R}$, $u \in C^{2} \left( B_{R}\right) \cap C \left( \overline{B}_{R} \right)$ --- субгармоническая в $B_{R}$ функция. Тогда
\begin{equation}
\label{2.22} u \left(x^0 \right) \leqslant \frac{1}{\omega_{n-1} R^{n-1}}\int \limits_{S_{R}}u(x) ds\footnote{Теорема о среднем по сфере для субгармонических функций.}.
\end{equation}
\end{theorem}

\begin{proof}
Пусть $0<\rho<R$. Фундаментальное решение можно модифицировать, положив $\tilde{E} \left( x-x^0 \right) = E \left( x-x^0 \right) + C$, чтобы $\tilde{E} < 0$ для всех $x \in B_{\rho}$, и $\tilde{E} = 0$, если $x \in S_{\rho}$ (найдите выражение для константы $C$). Для модифицированного фундаментального решения также верна формула интегрального представления. Действительно, $u \left( x^0 \right) =  \int_{S_{\rho} } \left( u \frac{\partial E}{\partial \vec{\nu}} - E \frac{\partial u}{\partial \vec{\nu}} \right) ds + \int_{B_{\rho}}  E \Delta u dx + \int_{B_{R}} C \Delta u dx - \int_{S_{\rho}} C  \frac{\partial u}{\partial \vec{\nu}} ds = \int_{S_{\rho} } \left( u \frac{\partial \tilde{E}}{\partial \vec{\nu}} - \tilde{E} \frac{\partial u}{\partial \vec{\nu}} \right) ds + \int_{B_{\rho}} \tilde{E} \Delta u dx =  \int_{S_{\rho} } \left( u \frac{\partial \tilde{E}}{\partial \vec{\nu}} - \tilde{E} \frac{\partial u}{\partial \vec{\nu}} \right) ds + \int_{B_{\rho}}  \tilde{E} \Delta u dx$. Так как $\frac{\partial \tilde{E}}{\partial \vec{\nu}}= \frac{\rho^{1-n}}{\omega_{n-1}}$ на $S_{\rho}$, из новой формулы интегрирования по частям имеем неравенство
\begin{equation}
\label{2.23} u \left(x^0 \right)  \leqslant  \frac{1}{\omega_{n-1} \rho^{n-1}}\int \limits_{S_{\rho}}u(x) ds.
\end{equation}
Так как $u \in C \left( \overline{B}_{R} \right)$, предельный переход $\rho \to R-0$ в неравенстве \eqref{2.23} даёт неравенство \eqref{2.22}
\end{proof}

\begin{theorem}\label{t.2.2}
Пусть $B_{R}  = \left\{ x \in \mathbb{R}^n: |x|<R\right\}$, $S_{R} = \partial B_{R}$, $u \in C^{2} \left( B_{R}\right) \cap C \left( \overline{B}_{R} \right)$ --- субгармоническая в $B_{R}$ функция. Тогда
\begin{equation}
\label{2.24} u \left(x^0 \right) \leqslant \frac{1}{\kappa_n R^{n}}\int \limits_{B_{R}}u(x) dx,
\end{equation}
где $\kappa_n = \frac{\omega_{n-1}}{n}$ --- объём шара единичного радиуса\footnote{Теорема о среднем по шару для субгармонических функций.}.
\end{theorem}

\begin{proof}
Умножим неравенство \eqref{2.23} на $\omega_{n-1} \rho^{n-1}$ и проинтегрируем по $\rho$ от $0$ до $R$: $u \left( x^0 \right) \int_{0}^{R} \omega_{n-1} \rho^{n-1} d \rho \leqslant \int_{0}^{R} d \rho \int_{S_{\rho}} u(x) ds$ или $u \left( x^0 \right) \kappa_{n}R^{n}  \leqslant \int_{B_{R}} u(x) dx$, что и требовалось\end{proof}

\begin{remark} \label{r.2.1}
Если $ u \in C^{2} \left( B_{R}\right) \cap C \left( \overline{B}_{R} \right)$ --- супергармоническая функция в шаре $B_{R}$ с центром в точке $x^0$, то поскольку $-u$ есть субгармоническая функция, то для $u$ справедливы неравенства, противоположные \eqref{2.22} и \eqref{2.24}. Гармонические функции одновременно суб- и супергармонические, поэтому для них справедливы теоремы о среднем в форме равенств: если $u \in C^{2} \left( B_{R}\right) \cap C \left( \overline{B}_{R} \right)$ --- гармоническая в $B_{R}$ функция, то
\begin{equation}
\label{2.25}
u \left(x^0 \right) =\frac{1}{\omega_{n-1} \rho^{n-1}}\int \limits_{S_{R}}u(x) ds,\,\, u \left(x^0 \right) = \frac{1}{\kappa_n R^{n}}\int \limits_{B_{R}}u(x) dx.
\end{equation}
\end{remark}

\begin{theorem}\label{t.2.3}
Пусть $\Omega \subset \mathbb{R}^n$ --- область, $u \in C^{2} (\Omega)$ --- субгармоническая функция, достигающая максимального значения $M$ в $\Omega$. Тогда $u \equiv M$ в $\Omega$\footnote{Сильный принцип максимума для субгармонических функций.}.
\end{theorem}

Теорема \ref{t.2.3} утверждает, что непостоянная субгармоническая функция в области $\Omega$ не может иметь внутреннего глобального максимума. Принцип максимума занимает центральное место в теории гармонических функций (а его обобщения --- в теории эллиптических уравнений второго порядка). Он, как многие ключевые утверждения в теории УЧП, имеет физическую интерпретацию. Рассмотрим трёхмерную область $\Omega$, заполненную сплошной средой со стационарным (не зависящем от времени) распределением температуры. Тогда либо температура во всех точках области $\Omega$ одинаковая, либо она не может иметь внутреннего максимума, иначе начнётся эволюция температуры.

\begin{proof}
Рассмотрим множество уровня $S_{M} = \{ x \in \Omega: u(x) = M\}$. По условию теоремы $S_M \ne \emptyset$, кроме того, $S_{M}$ --- замкнутое множество в топологии $\Omega$, потому что $u \in C(\Omega)$. Возьмём точку $x^0 \in S_{M}$, шар $B_{R} = B_{R} \left( x^0 \right) \Subset \Omega$ и применим неравенство со средним по шару для субгармонической функции $u -M$:
\begin{equation*}
0 = u \left( x^0 \right) - M \leqslant \frac{1}{\kappa_n R^{n}} \int \limits_{B_{R}} \left( u - M\right) dx.
\end{equation*}
С другой стороны, подынтегральная функция $u-M \leqslant 0$, поэтому $u  \equiv M$ в $B_{R}$, $B_{R} \subset S_{M}$ и множество $S_{M}$ открыто. Но область $\Omega$ топологически связна, единственное непустое открыто-замкнутое множество в $\Omega$ --- это сама область $\Omega$, т.е. $S_{M} = \Omega$, что и нужно было доказать \end{proof}

\begin{remark} \label{r.2.2}
Для супергармонических функций справедлив принцип минимума: если $u \in C^{2} (\Omega)$, $\Delta u \leqslant 0$, достигает минимум $m$ в $\Omega$, то $u \equiv m$ в $\Omega$. Гармонические функции удовлетворяют принципу максимума и минимума.
\end{remark}

\begin{corollary}\label{c.2.1}
Пусть $\Omega \Subset \mathbb{R}^n$, тогда: 1) непостоянная в $\Omega$ субгармоническая функция $u \in C^{2} (\Omega) \cap C \left( \overline{\Omega}\right)$ удовлетворяет в любой точке $x \in \Omega$ неравенству $u(x) < \max \limits_{\partial \Omega} u$; 2) непостоянная в $\Omega$ супергармоническая функция $u \in C^{2} (\Omega) \cap C \left( \overline{\Omega}\right)$ удовлетворяет в любой точке $x \in \Omega$ неравенству $u(x) > \min \limits_{\partial \Omega} u$; 3) непостоянная в $\Omega$ гармоническая функция $u \in C^{2} (\Omega) \cap C \left( \overline{\Omega}\right)$ удовлетворяет в любой точке $x \in \Omega$ двойному неравенству
\begin{equation}
\label{2.26} \min \limits_{\partial \Omega} u < u(x) < \max \limits_{\partial \Omega} u.
\end{equation}
\end{corollary}

\begin{proof}
Так как $\overline{\Omega}$ --- компакт, непрерывная на нём субгармоническая функция $u$ достигает максимального значения. Если точка максимума $x^0$ принадлежит $\Omega$, то по теореме \ref{t.2.3} $u \equiv u \left( x^0 \right)$, что противоречит условию. Следовательно максимум достигается на границе, а во внутренних точках $u(x) < \max_{\partial \Omega} u$. Аналогично для супергармонической функции $u$ в каждой $x \in \Omega$ $u(x) > \min_{\partial \Omega} u$. Третий пункт следует из первых двух\end{proof}

Пусть $\Omega \subset \mathbb{R}^n$ --- область, $\varphi: \partial \Omega \to \mathbb{R}$ --- известная функция на границе (граничная функция). Задача Дирихле для уравнения Лапласа в области $\Omega$ имеет вид
\begin{equation}
\label{2.27}
\left\{
\begin{aligned}
& \Delta u =0 \, \mbox{в}\,\Omega,\\
& \left. u \right|_{\partial \Omega} = \varphi,
\end{aligned}
\right.
\end{equation}
где $\left. u \right|_{\partial \Omega} = \varphi$ означает, что $u(x) = \varphi(x)$ для всех $x \in \partial \Omega$. Задача \eqref{2.27} называется внутренней, если $\Omega$ --- ограниченная область.

\begin{definition} \label{d.2.9}
Классическим решением задачи Дирихле \eqref{2.27} называется функция $u \in C^{2} (\Omega) \cap C \left( \overline{\Omega} \right)$, которая при подстановке в задачу обращает уравнение Лапласа и условие Дирихле в тождества.
\end{definition}

Очевидно, необходимое условие существования классического решения --- это $\varphi \in C (\partial \Omega)$, так как $u$ не только должна совпадать с $\varphi$ на границе, но и быть непрерывной в $\overline{\Omega}$. В функциональной теории УЧП в первую очередь исследуется корректность поставленных (краевых, начальных, начально-краевых и других) задач. Мы дадим определение корректно поставленных по Адамару линейных задач\footnote{Определение в нелинейном случае аналогичное, но будет приведено отдельно для каждой задачи.}, проходящее красной нитью через всю книгу, имея в качестве примера внутреннюю задачу \eqref{2.27}.

Итак, пусть $\mathcal{U}$ --- пространство\footnote{Под ‘пространством’ по умолчанию понимается функциональное пространство, замкнутое относительно естественных линейных операций.} решений (в том или другом смысле), $\mathcal{F}$ --- пространство правых частей (данных) линейной задачи. В пространствах $\mathcal{U}$, $\mathcal{F}$ задана топология или хотя бы фиксирован класс сходящихся последовательностей. Линейной задаче соответствует линейный оператор $\mathcal{U} \to \mathcal{F}$, возможно определённый не на всём $\mathcal{U}$. Например, $\mathcal{U} = C^{2} (\Omega) \cap C \left( \overline{\Omega} \right)$ --- пространство классических решений, $\mathcal{F} = C (\partial \Omega)$ --- пространство данных задачи Дирихле \eqref{2.27}. Сходимость последовательности $\{ \varphi_{k} \}_{k=1}^{\infty} \subset \mathcal{F}$ --- это сходимость относительно равномерной нормы $\| \varphi \| = \max_{\partial \Omega} |\varphi|$; последовательность $\{ u_{k} \}_{k=1}^{\infty} \subset \mathcal{U}$ сходится, если она сходится в $C \left( \overline{\Omega} \right)$ и равномерно сходится вместе со всеми производными до второго порядка включительно на любом компакте из $\Omega$.

\begin{definition} \label{d.2.10}
Линейная задача называется корректно поставленной по Адамару (или, просто, корректной) для пространств $\mathcal{U}$, $\mathcal{F}$, если выполнены три положения:
\begin{enumerate}
\item[1)] решение существует для любых данных из $\mathcal{F}$ (существование);

\item[2)] решение единственно в пространстве $\mathcal{U}$ (единственность);

\item[3)] разрешающий оператор $R: \mathcal{F} \to \mathcal{U}$, сопоставляющий правым частям из $\mathcal{F}$ единственное решение из $\mathcal{U}$, непрерывен (устойчивость).
\end{enumerate}
\end{definition}

Начав со второго пункта определения \ref{d.2.10}, т.е. с единственности, до конца этой главы мы докажем корректность задачи Дирихле \eqref{2.27} для ограниченных областей с так называемыми регулярными границами, причём пространство решений будет сужено до $C^{\infty}(\Omega) \cap C \left( \overline{\Omega} \right)$. Заметим, что единственность решения линейной краевой задачи эквивалентна единственности тривиального решения задачи с нулевыми данными. Для внутренней задачи Дирихле она имеет вид
\begin{equation}
\label{2.28}
\left\{
\begin{aligned}
& \Delta u =0 \, \mbox{в}\,\Omega \Subset \mathbb{R}^n,\\
& \left. u \right|_{\partial \Omega} = 0.\\
\end{aligned}
\right.
\end{equation}

\begin{corollary}\label{c.2.2}
Классическое решение внутренней задачи Дирихле для уравнения Лапласа единственно.
\end{corollary}

\begin{proof}
Пусть $u_1, u_2 \in C^2 (\Omega) \cap C \left( \overline{\Omega}\right)$ --- два решения задачи \eqref{2.27}. Тогда их разность $u=u_1-u_2$ удовлетворяет задаче \eqref{2.28}. Если $u$ не равна нулю тождественно, то в любой $x \in \Omega$ по следствию \ref{c.2.1} имеем $0 =\min_{\partial \Omega} u < u(x) < \max_{\partial \Omega} u =0$ --- противоречие. Отсюда $u_1 \equiv u_2$ в $\Omega$ \end{proof}

\section{Качественная теория гармонических функций}
\markright{\thesection\ Качественная теория гармонических функций}

Если функциональная теория уравнений в частных производных изучает разрешимость задач и основные свойства решений, то качественная теория занимается исследованием тонких свойств решений, зачастую не представляя само решение в каком-либо в виде, т.е. не используя тот или иной метод построения.<...>

Для доказательства бесконечной дифференцируемости гармонических функций мы разовьём технику усреднения, хотя её область применения значительно шире. Суть усреднения --- это сглаживание функций, причём удовлетворяющее ряду замечательных свойств. Речь идёт об операторе $J_{h}: L_{1,loc}  \left( \mathbb{R}^n \right) \to C^{\infty} \left( \mathbb{R}^n \right)$, где чем меньше положительный параметр $h$, называемым радиусом усреднения, тем ближе (в некотором смысле) функция к своему усреднению. Для определения оператора $J_{h}$ нам будет нужна одна основная функция.

\begin{definition} \label{d.2.11}
Пусть $h>0$. Ядром усреднения радиуса $h$ называется функция $\zeta_{h} \in \mathcal{D} \left( \mathbb{R}^n \right)$, удовлетворяющая условиям: 1) $\zeta_{h} \geqslant 0$; 2) $\zeta_{h}$ --- радиально-симметричная функция; 3) $\mathrm{supp}\, \zeta_{h} = \overline{B}_{h}(0)$; 4) $\int_{B_{h}(0)} \zeta_{h}(x) dx =1$.
\end{definition}

Примером ядра усреднения может служить функция
\begin{equation*}
\zeta_{h}(x) = \left\{
\begin{aligned}
& \frac{C_{n}}{h^n} \exp \left( \frac{-h^2}{h^2 - |x|^2} \right),\, |x|<h,\\
& 0,\, |x| \geqslant h,\\
\end{aligned}
\right.
\end{equation*}
где положительная константа $C_{n}$, зависящая только от размерности, подобрана так, чтобы указанная функция удовлетворяла четвёртому пункту в определении \ref{d.2.11} (первым трём, как нетрудно убедиться, она уже удовлетворяет). Пусть теперь $\zeta_{h}$ --- любое ядро усреднения радиуса $h$, $u \in L_{1,loc} \left( \mathbb{R}^n \right)$, тогда результатом усреднения будем функция
\begin{equation}
\label{2.29}
\left( J_{h} u \right) (x) = \int \limits_{B_{h} (x)} \zeta_{h} (x-y) u(y) dy,
\end{equation}
определённая для всех $x \in \mathbb{R}^n$. Интеграл в формуле \eqref{2.29} называется свёрткой и обозначается $\zeta_{h} * u$. Как будет показано в четвёртой главе, свёртка любой обобщённой функции с основной (финитной бесконечно дифференцируемой) функцией существует, бесконечно дифференцируема и для любого $\alpha \in \mathbb{Z}_{+}^{n}$ $D^{\alpha}  \left( \zeta_{h} * u \right) = \left( D^{\alpha} \zeta_{h} \right) * u$. 
Поэтому функция, заданная формулой \eqref{2.29}, бесконечно дифференцируема, и мы определили оператор усреднения $J_{h}:  L_{1,loc}  \left( \mathbb{R}^n \right) \to C^{\infty} \left( \mathbb{R}^n \right)$. Его можно рассматривать и в других функциональных пространствах, например, в $L_{p}(\Omega)$, где $\Omega$ --- ограниченная область. Договоримся продолжать функции, определённые в $\Omega$, нулём в дополнении $\Omega$ до $\mathbb{R}^n$. Основное свойство оператора усреднения состоит в том, что для любых $\Omega \Subset \mathbb{R}^n$, $u \in L_{p}(\Omega)$  $\| J_{h} u\|_{L_{p}(\Omega)} \leqslant \| u \|_{L_{p}(\Omega)}$ и $\| u-J_{h} u \|_{L_{p}(\Omega)} \to 0$ при $h \to 0$. Это утверждение будет доказано в следующих главах книги, а сейчас мы докажем, что гармоническая функция в $\Omega$ принадлежит $C^{\infty}(\Omega)$.

\begin{proposition}\label{p.2.3}
Функция $u \in C^{2}(\Omega)$, гармоническая в области $\Omega \subset \mathbb{R}^n$, бесконечно дифференцируема в $\Omega$.
 \end{proposition}
 
 \begin{proof}
Фиксируем $\varepsilon >0$, рассмотрим открытое подмножество $Q_{\varepsilon} = \left\{ x \in \Omega: \mathrm{dist} (x, \partial \Omega) > \varepsilon \right\}$ (считаем $\varepsilon$ достаточно малым, что $Q_{\varepsilon}$ непусто) и пусть $\rho < \varepsilon$. Так как $u$ --- гармоническая функция, её значение любой точке $x \in Q_{\varepsilon}$ равно среднему по сфере $S_{\rho}(x)$:
\begin{equation}
\label{2.30} u  (x) = \frac{1}{\omega_{n-1} \rho^{n-1}}\int \limits_{S_{\rho} (x)}u(y) ds.
\end{equation}
Выберем радиус усреднения $h<\varepsilon$, умножим \eqref{2.30} на $\omega_{n-1} \rho^{n-1} \zeta_{h} (\rho)$ и проинтегрируем по $\rho$ от $0$ до $h$: $u(x)=u (x) \int \limits_{0}^{h} \omega_{n-1} \rho^{n-1} \zeta_{h} (\rho) d \rho =$
\begin{equation*}
=  \int \limits_{0}^{h} \zeta_{h}(\rho) d \rho \int \limits_{S_{\rho} (x) } u(y) ds = \int \limits_{B_{h}(x)} \zeta_{h} (x-y) u(y) dy = \left( \mathrm{J}_{h} u \right) (x).
\end{equation*}
Таким образом, функция $u$ совпадает со своим усреднением радиуса $h<\varepsilon$ в области $Q_{\varepsilon}$. Так как $J_{h} u$ --- гладкая функция в этой области, $u \in C^{\infty} \left( Q_{\varepsilon}\right)$ при любом $\varepsilon>0$. Устремляя $\varepsilon$ к нулю, имеем $u \in C^{\infty}(\Omega)$ \end{proof}

В силу коммутационного соотношения $\Delta D^{\alpha} = D^{\alpha} \Delta$ все производные гармонических функций --- тоже гармонические функции. Предположим, что некоторая бесконечно дифференцируемая функция ограничена, т.е. существует константа $C$ и $|u(x)| \leqslant C$ для всех $x$ из области $\Omega$. Что можно сказать о производных функции $u$? Ровным счётом ничего: оценка значений функции не даёт какой-либо оценки её производных\footnote{Оценка производных тем не менее даёт некоторую информацию о самой функции, например, в теореме Лагранжа о конечных приращениях или в неравенстве Фридрихса.}. Ситуация резко меняется, когда рассматриваемая функция есть решение уравнения в частных производных: теперь оценить производные становится возможным. Соответствующие оценки называют в качественной теории УЧП априорными. Априорные оценки позволяют оценить решение уравнения и/или его производные без предъявления какой-либо явной формулы для решения. Они бывают разных видов (точечные, интегральные, внутренние, глобальные), и первой в этой книге будет доказана внутренняя точечная априорная оценка производных гармонических функций.

\begin{theorem}\label{t.2.4}
Пусть $u \in C^{2}(\Omega) \cap C \left( \overline{\Omega}\right)$ --- гармоническая функция в области $\Omega \Subset \mathbb{R}^n$, подобласть $Q \Subset \Omega$. Тогда для любого порядка $k \in \mathbb{N}$ и любой точки $x^0 \in \overline{Q}$ выполнено неравенство
\begin{equation}
\label{2.31} \left| D^{\alpha}  \left( x^0 \right)  \right| \leqslant \left( \frac{nk}{d}\right)^{k} \max \limits_{ \partial \Omega}|u|,
\end{equation}
где $\alpha \in \mathbb{Z}_{+}^{n}$, $|\alpha| = k$, $d = \mathrm{dist}(Q, \partial \Omega)$\footnote{Это кратчайшее растояние, а не расстояние Хаусдорфа: $\mathrm{dist}(Q, \partial \Omega)=\inf \limits_{x \in Q} \mathrm{dist}(x,\partial \Omega)$.}
\end{theorem}

 \begin{proof}
Пусть $B_{R}  \left( x^0 \right)  \Subset \Omega$, тогда по теореме о среднем по шару и формуле интегрирования по частям
\begin{equation*}
u_{x_j} \left( x^0 \right) = \frac{1}{\kappa_n R^n}\int \limits_{B_{R}\left( x^0 \right)} \frac{\partial u}{\partial x_j} dx=\frac{1}{\kappa_n R^n} \int \limits_{S_{R}\left( x^0 \right)} u \nu_{j} ds.
\end{equation*}
Оцениваем и применяем принцип максимума:
\begin{equation*}
\left| u_{x_j} \left( x^0 \right) \right| \leqslant \frac{\omega_{n-1} R^{n-1}}{\kappa_n R^n } \max \limits_{S_{R} \left( x^0 \right)}|u| = \frac{n}{R} \max \limits_{S_{R} \left( x^0 \right)}|u|.
\end{equation*}
Пусть теперь $\alpha \in \mathbb{Z}_{+}^{n}$, $|\alpha| = k \geqslant 1$. Построим семейство подобластей $Q = \Omega_1  \Subset \Omega_2 \Subset \ldots \Subset \Omega_{k+1} = \Omega$, где
$\Omega_{j+1} =  \left\{ x \in \Omega: \mathrm{dist} \left(x, \Omega_j \right) < \frac{d}{k}\right\}$, $j=1,\ldots,k-1$. Для любой точки $x^0 \in \Omega_j$ шар $B_{\frac{d}{k}} \left( x^0 \right)\Subset \Omega_{j+1}$.
Теперь, используя первую часть доказательства, рассмотрим $x^0 \in \overline{Q}$ и запишем цепочку неравенств
\begin{equation*}
\left| D^{\alpha} u \left( x^0 \right) \right| \leqslant \frac{nk}{d} \max \limits_{\overline{\Omega}_2} \left| D^{\alpha^1} u  \right| \leqslant \ldots \leqslant  \left( \frac{nk}{d}\right)^k \max \limits_{\overline{\Omega}} |u| = \left( \frac{nk}{d}\right)^k \max \limits_{\partial \Omega} |u| ,
\end{equation*}
где $\left| \alpha^{j+1}\right|  = \left| \alpha^{j}\right| - 1$, $\alpha^{j+1}$ отличается от $\alpha^j$ одним индексом. Неравенство \eqref{2.31} доказано \end{proof}

Априорная оценка производных гармонических следствий имеет ряд замечательных следствий, но в первую очередь мы получим результат об устойчивости внутренней задачи Дирихле \eqref{2.27}. Вспомним, что норма в пространстве $C^{k} \left( \overline{Q} \right)$ задаётся нормой $\| u \|_{k, \overline{Q}} = \sum \limits_{|\alpha| \leqslant k} \max \limits_{\overline{Q}} \left| D^{\alpha} u \right|$, и относительно этой нормы пространство $C^{k} \left( \overline{Q} \right)$ банахово. В следующем определении фиксируется класс сходящихся последовательностей в пространстве решений $\mathcal{U} = C^{\infty}(\Omega) \cap C \left( \overline{\Omega} \right)$.

\begin{definition} \label{d.2.12}
Пусть $\Omega \subset \mathbb{R}^n$ --- область (не обязательно ограниченная). Последовательность $\left\{ u_{j} \right\}_{j=1}^{\infty} \subset C^{\infty}(\Omega)$ сходится к $u \in C^{\infty}(\Omega)$, если для любой $Q \Subset \Omega$ и любого $k \in \mathbb{N}_0$ $\left\| u_{j} - u \right\|_{k,\overline{Q}} \to 0$ при $j \to \infty$. Для ограниченной области $\Omega$ последовательность  $\left\{ u_{j} \right\}_{j=1}^{\infty} \subset \mathcal{U}$ сходится к $u \in \mathcal{U}$, если она сходится к этой функции в $C \left( \overline{\Omega} \right)$ и в $C^{\infty}(\Omega)$.
\end{definition}

\begin{proposition}\label{p.2.4}
Предположим, что последовательность граничных функций $\left\{ \varphi_j \right\}_{j=1}^{\infty} \subset C(\partial \Omega)$ сходится к функции $\varphi \in C(\partial \Omega)$ и для каждой $\varphi_j$ существует решение задачи \eqref{2.27} $u_{j} \in \mathcal{U}$; кроме того, существует решение $u \in \mathcal{U}$ для предельной функции $\varphi$. Тогда $u_j \to u$ при $j \to \infty$ в $\mathcal{U}$.
\end{proposition}

 \begin{proof}
Сходимость последовательности $\left\{ u_{j} \right\}_{j=1}^{\infty}$ в $C \left( \overline{\Omega} \right)$ следует непосредственно из принципа максимума:
\begin{equation*}
\left\| u_j - u \right\|_{C \left( \overline{\Omega} \right)} = \max_{\overline{\Omega}} \left| u_j - u \right| = \max_{\partial \Omega} \left| \varphi_j - \varphi \right|  = \left\| \varphi_j - \varphi \right\| \to 0, j \to \infty.
\end{equation*}
Фиксируем подобласть $Q \Subset \Omega$, число $k \in \mathbb{N}$, получим оценку $\left\| u_j - u \right\|_{k,\overline{Q}}$ ($d = \mathrm{dist}(Q, \partial \Omega)$):
\begin{equation*}
\left\| u_j - u \right\|_{k,\overline{Q}} = \sum \limits_{|\alpha| \leqslant k} \left\| u_j - u \right\|_{C \left( \overline{Q} \right)} \leqslant \sum \limits_{|\alpha| \leqslant k} \left( \frac{n |\alpha|}{d} \right)^{|\alpha|} \max \limits_{\partial \Omega} \left\| \varphi_j - \varphi \right\|  \to 0,\,\, j \to \infty.
\end{equation*}
Следовательно, $\left\{ u_{j} \right\}_{j=1}^{\infty}$ сходится в $C^{k} \left( \overline{Q} \right)$, а так как $Q$ и $k$ произвольны, эта последовательность сходится в $\mathcal{U}$ \end{proof}

Таким образом, для корректности по Адамару внутренней задачи Дирихле \eqref{2.27} осталось доказать существование решения (первый пункт в определении \ref{d.2.10}), что составляет наибольшую сложность, но будет сделано в следующем параграфе. Здесь продолжим изучение основных и тонких свойств гармонических функций. Оказывается, всякая гармоническая в области $\Omega$ функция $u$ допускает представление в окрестности любой точки $x^0 \in \Omega$ в виде ряда по степеням $x_1-x_1^0,\ldots,x_n-x_n^0$, т.е. $u$ --- аналитическая в $\Omega$ функция.

\begin{theorem}\label{t.2.5}
Гармонические функции аналитичны в своей области определения.
\end{theorem}

\begin{proof}
Пусть $u$ --- гармоническая в $\Omega$ функция, тогда она бесконечно дифференцируема в $\Omega$ (по утверждению \ref{p.2.3}). Рассмотрим $x^0 \in \Omega$ и пусть радиус $\rho>0$ таков, что $B_{2 \rho} \left( x^0 \right) \subset \Omega$. Пусть $x \in B_{\rho} \left( x^0 \right)$, запишем формулу Тейлора:
\begin{equation}
\label{2.32} u(x) = \sum \limits_{|\alpha| < m} \frac{1}{\alpha!} D^{\alpha} u \left( x^0 \right) \left( x-x^0 \right)^{\alpha} + R_{m} \left( x; x^0 \right),
\end{equation}
в которой $R_{m}$ --- остаточный член. Функция $u$ представляется рядом Тейлора $\sum_{\alpha} \frac{1}{\alpha!} D^{\alpha} u \left( x^0 \right) \left( x-x^0 \right)^{\alpha}$ в окрестности $x^0$, если существует $\varepsilon>0$,   $R_{m} \to 0$ при $m \to \infty$ равномерно по $x \in \overline{B}_{\varepsilon} \left( x^0 \right)$, $R_{m} \to 0$. Представим остаточный член в форме Лагранжа:
\begin{equation}
\label{2.33}
R_{m}  \left( x;x^0 \right) = \sum \limits_{|\alpha|=m}  \frac{1}{\alpha!} D^{\alpha} u \left( x' \right) \left( x-x^0 \right)^{\alpha},
\end{equation}
где $x \in B_{\rho} \left( x^0 \right)$, $x' = (1-t) x^0 + t x$, $t \in (0,1)$. Из теоремы \ref{t.2.4} явствует неравенство  $\left|  D^{\alpha} u \left( x' \right) \right| \leqslant \left( \frac{nm}{\rho} \right)^{m} \max \limits_{\overline{B}_{2 \rho} \left( x^0 \right)} |u|$. Кроме него, для оценки остаточного члена \eqref{2.33}, нам потребуется тождество мультиномиальных коэффициентов $\sum_{|\alpha|=m} \frac{m!}{\alpha! }= n^m$ и неравенство Стирлинга $m! \geqslant \left( \frac{m}{e} \right)^{m}$, откуда $\sum_{|\alpha|=m} \frac{1}{\alpha!} \leqslant \left( \frac{ne}{m} \right)^{m}$. Имеем
\begin{equation*}
\left| R_{m}  \left( x;x^0 \right)  \right| \leqslant  \left( \frac{nm}{\rho} \right)^{m} \max \limits_{\overline{B}_{2 \rho} \left( x^0 \right)} |u|  \left| x-x^0 \right|^{m} \sum \limits_{|\alpha|=m} \frac{1}{\alpha!} \leqslant \left( n^2 \left| x-x^0 \right| e \rho^{-1} \right)^{m} \max \limits_{\overline{B}_{2 \rho} \left( x^0 \right)} |u|.
\end{equation*}
Тогда возьмём $\varepsilon < \frac{\rho}{n^2 e}$ и получим: $R_m \left( x,x^0 \right) \to 0$, $m \to \infty$, равномерно по $x \in \overline{B}_{\varepsilon} \left( x^0 \right)$, что и нужно было для аналитичности функции $u$\end{proof}

Качественная теория гармонических функций отчасти повторяет теорию голоморфных функций одной комплексной переменной, но если в последней основным инструментом доказательства бесконечной дифференцируемости и разложимости в степенные рядя Тейлора и Лорана служил интеграл Коши $\oint_{\Gamma} \frac{f(z)}{z-z_0} dz$ вдоль замкнутой кривой жордана $\Gamma$, то его роль в теории гармонических функций выполняют теоремы о среднем \ref{t.2.1} и \ref{t.2.2} в сочетании с принципом максимума\footnote{В ТФКП, впрочем, есть принцип максимума модуля аргумента: если модуль голоморфной функции $f$ достигает максимум в области $\Omega$, то $f = \mathrm{const}$ в $\Omega$.}. Следующая теорема  обобщают соответствующее утверждение из ТФКП.

\begin{theorem}\label{t.2.6}
Пусть $u$ --- гармоническая функция в $\mathbb{R}^n$, существуют такие константы $m \geqslant 0$, $C \geqslant 0$, что для любой $x \in \mathbb{R}^n$ выполнено неравенство
\begin{equation}
\label{2.34} |u(x)| \leqslant C \left( 1+ |x|^m \right).
\end{equation}
Тогда $u$ --- многочлен от $x_1,\ldots,x_n$ со степенью $\deg u \leqslant [m]$\footnote{Теорема Лиувилля для гармонических функций.}. В частности, если гармоническая в евклидовом пространстве функция ограничена, то она постоянна.
\end{theorem}

\begin{proof}
Рассмотрим шары $B_{R+\rho} $, $B_{R} $ с центром в начале координат, где $R, \rho>0$. Пусть $\alpha \in \mathbb{Z}_{+}^{n}$, $|\alpha| =  k = [m]+1$. Априорная оценка даёт $\max_{\overline{B}_{R}} \left| D^{\alpha} u\right| \leqslant  \left( \frac{nk}{\rho}\right)^k \max_{\overline{B}_{R+\rho}} |u| \leqslant (nk)^k \rho^{-k} C \left( 1+ |R+\rho|^m\right)$. Так как $k>m$, то при $\rho \to +\infty$ и фиксированном $R$ правая часть последнего неравенства сттремится к нулю. Следовательно, при любом $R>0$ и любом $\alpha \in \mathbb{Z}_{+}^{n}$, $|\alpha|=[m]+1$, $\max_{\overline{B}_{R}}\left| D^{\alpha} u\right| = 0$, т.е. $D^{\alpha} u =0$ в $\mathbb{R}^n$. Отсюда следует, что $u$ --- многочлен степени не больше, чем $[m]$\end{proof}

\section{Разрешимость задачи Дирихле для уравнения Лапласа}
\markright{\thesection\ Разрешимость задачи Дирихле для уравнения Лапласа}

Совершенно ясно, что существование решения внутренней задачи Дирихле для уравнения Лапласа \eqref{2.27} сильно зависит от границы области $\Omega$: для некоторых <<хороших>> границ решение существует при любой функции $\varphi \in C(\partial \Omega)$, для других --- только если граничная функция принадлежит некоторому подпространству $C(\partial \Omega)$ (заметим, что для $\varphi =0$ всегда существует тривиальное решение). Установление класса <<хороших>> границ --- одна из главных задач этого параграфа. Кроме того, будет предложен метод построения решения задачи \eqref{2.27}, правда, не имеющий практических приложений. В функциональной теории УЧП есть два пути доказательства разрешимости. Первый (прямой) путь основан на том или ином методе построения решения. Чаще всего решение определяется как предел по некоторому фильтру, в самых простых случаях (например, в методе Фурье) --- как предел последовательности (ряд). Тогда исследователь должен получить результат о сходимости полученного ряда в соответствующем функциональном пространстве $\mathcal{U}$ к решению (в указанном смысле) краевой задачи. Второй (топологический) путь основан на применении той или другой теоремы о неподвижной точке. Например, принципа Шаудера, который утверждает, что если нелинейный оператор, отображающий шар банахового пространства в себя, компактен, то у этого оператора есть неподвижная точка. Или метода монотонных операторов, который будет использован для доказательства разрешимости задачи Дирихле для нелинейного эллиптического уравнения второго порядка в следующих главах книги.

Однако, мы начнём с доказательства разрешимости внутренней задачи Дирихле в идеальной (с точки зрения оператора Лапласа) области --- шаре. Для решения задачи Дирихле в шаре есть явная формула (интеграл Пуассона), происхождение которой мы объясним в третьей главе. Итак, пусть $B_{R} = \left\{ x \in \mathbb{R}^n: |x|<R \right\}$ --- шар с радиусом $R$, $S_{R} = \partial B_{R}$; рассмотрим задачу
\begin{equation}
\label{2.5.1}
\left\{
\begin{aligned}
& \Delta u =0 \, \mbox{в}\,B_R,\\
& \left. u \right|_{S_{R}} = \varphi.
\end{aligned}
\right.
\end{equation}

\begin{theorem}\label{t.2.5.1}
Пусть $\varphi \in C \left( S_R \right)$, тогда функция 
\begin{equation}
\label{2.5.2} u(x) = \frac{R^2 - | x |^2}{ \omega_{n-1} R} \int \limits_{S_{R}} \frac{\varphi(y) ds}{ \left| x - y \right|^{n}},  x \in B_{R},
\end{equation}
гармоническая в $B_{R}$, принадлежит $C \left( \overline{B}_{R} \right)$, и для любой точки $x^0 \in S_{R}$ $\lim \limits_{B_{R} \ni x \to x^0}u(x) = \varphi \left( x^0\right)$. Другими словами, функция $u$ есть классическое решение задачи \eqref{2.5.1}.
\end{theorem}

\begin{proof}
Заметим, что для любой точки $x^0 \in B_{R}$ интеграл \eqref{2.5.2} собственный, поэтому из бесконечной дифференцируемости интегрального ядра $ \frac{R^2 - | x |^2}{ \omega_{n-1} R} \frac{1}{|x-y|^n}$ следует, что $u \in C^{\infty} \left( B_R \right)$. Проверим, что $u$ --- гармоническая функция. Обозначим $\xi = y+\xi$, $\rho = |\xi|$, представим ядро Пуассона в виде $K = \frac{1}{\omega_{n-1}R} \left( R^2 - |y+\xi|^2 \right) \rho^{-n}$. Используем, во-первых, формулу для оператора Лапласа от радиально-симметричных функций $\Delta v(\rho) = v''(\rho) + \frac{n-1}{\rho} v'(\rho)$, во-вторых, формулу для оператора Лапласа от произведения функций $\Delta (uv) = u \Delta v + 2 \nabla u \cdot \nabla v + v \Delta u$. Кроме того, $|y+\xi|^2 = R^2 + 2 \vec{y} \cdot \vec{\xi} + \rho^2 $. Имеем
\begin{equation*}
\begin{aligned}
& \Delta K  = -\frac{2 n}{\omega_{n-1} R } \rho^{-n}  + \frac{4n}{\omega_{n-1}R} \rho^{-n-2} \sum_{j=1}^{n} \left( y_j + \xi_j \right) \xi_j +\frac{R^2 - |y+\xi|^2}{\omega_{n-1}R} 2 n \rho^{-n-2} =  \\ 
&  = \frac{2n}{\omega_{n-1} R} \rho^{-n-2} \left( -\rho^2 + 2 \vec{y} \cdot \vec{\xi} + 2 \rho^2  + R^2 - R^2 -  2 \vec{y} \cdot \vec{\xi} - \rho^2\right) = 0.\\
\end{aligned}
\end{equation*} 
Отсюда $\Delta u = \int_{S_R} \varphi(y) \Delta K ds =0$.

Пусть $\varphi \equiv 1$, тогда задача  \eqref{2.5.1} имеет очевидное решение $u \equiv 1$, и оно единственно по следствию \ref{c.2.2} Запишем для него формулу интегрального представления \eqref{2.11}, учитывая то, что нормальная производная для сферы $S_{R}$ есть производная по сферическому радиусу $r=|y|$:
\begin{equation*}
\begin{aligned}
& 1= \int \limits_{S_{R}} \partial_{\nu} \mathcal{E}\left( | x-y| \right) ds = \int \limits_{S_{R}} \left. \frac{n-2}{\omega_{n-1} (n-2)} |x-y|^{1-n} \frac{r - |x| \cos \gamma}{|x-y|} \right|_{r=R} ds =\\
& = \int \limits_{S_{R}} \frac{R^2 - R |x| \cos \gamma}{R \omega_{n-1} |x-y|^n} ds = \frac{R^2 - |x|^2}{ \omega_{n-1} R} \int \limits_{S_{R}} \frac{ds}{|x-y|^n}.\\
\end{aligned}
\end{equation*}

Пусть теперь $\varphi \in C \left( S_{R} \right)$, покажем, что для любой точки $x^0 \in S_{R}$ и любого $\varepsilon >0$ найдётся $\delta >0 $ со свойством: если $x \in B_{R} \cap B_{\delta} \left(x^0 \right)$, то $\left| u(x) - \varphi \left(x^0 \right) \right| < \varepsilon$. Поскольку $\varphi$ --- непрерывная функция на $S_R$, существует $\delta' >0$, что для всех $y \in S_{R} \cap B_{\delta'} \left(x^0 \right)$ справедливо неравенство $\left| \varphi \left( x^0 \right) - \varphi(y) \right| < \frac{\varepsilon}{2}$. Обозначим через $\sigma_{\delta'}$ пересечение сферы $S_{R}$ с шаром $B_{\delta'}\left( x^0 \right)$ и рассмотрим оценку:
\begin{equation*}
\begin{aligned}
& \left| u(x) - \varphi \left( x^0 \right) \right|  =  \left| \int \limits_{S_{R}} K \left( y,x \right) \left( \varphi(y) - \varphi \left( x^0 \right) \right) ds\right| \leqslant \int \limits_{\sigma_{\delta'}} K \left( y,x \right) \left| \varphi(y) - \varphi \left( x^0 \right)  \right| ds + \\
& + \int \limits_{S_{R} \backslash \sigma_{\delta'}} K \left( y,x \right) \left| \varphi(y) - \varphi \left( x^0 \right)  \right| ds < \frac{\varepsilon}{2} + \sup \limits_{y \in S_{R} \backslash \sigma_{\delta'}} \left| \varphi(y) - \varphi \left( x^0 \right)  \right| \int \limits_{S_{R} \backslash \sigma_{\delta'}} K(x,y) dy.\\
\end{aligned}
\end{equation*}
Если $M = \max_{S_{R}} | \varphi|$, то $\sup_{y \in S_{R} \backslash \sigma_{\delta'}} \left| \varphi(y) - \varphi \left( x^0 \right)  \right|  \leqslant 2 M$. Осталось оценить интеграл $\int_{S_{R} \backslash \sigma_{\delta'}} K(x,y) dy$:
\begin{equation*}
\int \limits_{S_{R} \backslash \sigma_{\delta'}} K(x,y) dy = \frac{R^2 - |x|^2}{\omega_{n-1}R} \omega_{n-1} R^{n-1}  \frac{1}{\left| \delta' - \mu \right|^n} \leqslant \frac{2 R^{n-1}  \left( R - |x| \right)}{\left| \delta' - \mu \right|^n}
\end{equation*}
при  $\left| x - x^0 \right| < \mu < \delta'$. Если, кроме того, $R-\varepsilon \frac{\left| \delta' - \mu \right|^n}{ 8M R^{n-1}}<|x|<R$ (рис. 2.2), то оценку можно продолжить и получить $\int_{S_{R} \backslash \sigma_{\delta'}} K(x,y) dy\leqslant \frac{\varepsilon}{4 M}$. Следовательно, положив, к примеру $\mu = \frac{1}{2} \delta'$, для $\delta = \frac{\varepsilon \left( \delta' \right)^n}{2^{n+3} M R^{n-1}}$ и любой $x \in B_{R} \cap B_{\delta'} \left( x^0 \right)$ будем иметь $\left| u(x) - \varphi \left( x^0 \right) \right| < \frac{\varepsilon}{2} +2 M \frac{\varepsilon}{4M} < \varepsilon$, что и требовалось \end{proof}

Формула \eqref{2.5.2} даёт не только решение задачи Дирихле в шаре; из неё также следует обратная теорема о среднем, не говоря уже о том, на формуле Пуассона основан метод построения гармонической функции в произвольной области, который будет изложен дальше.

\begin{theorem}\label{t.2.5.2}
Пусть $\Omega$ --- область в $\mathbb{R}^n$, $u \in C(\Omega)$ и для любой сферы $S_{R} \left( x^0 \right) \subset \Omega$ выполнено соотношение
\begin{equation}
\label{2.5.3}
u \left(x^0 \right) = \frac{1}{\omega_{n-1} R^{n-1}}\int \limits_{S_{R} \left( x^0 \right)}u(x) ds, 
\end{equation}
т.е. значение функции $u$ в точке $x^0$ равно среднему по любой сфере с центром в этой точке. Тогда $u$ --- гармоническая функция в $\Omega$\footnote{Обратная теорема о среднем для гармонических функций.}.
\end{theorem}

\begin{proof}
Пусть $\varphi =  \left. u \right|_{S_{R} \left( x^0 \right)}$, применим к этой функции и сфере $S_{R} \left( x^0 \right)$ формулу Пуассона (со сдвинутым в точку $x^0$ центром):
\begin{equation}
\label{2.5.4}
v(x) = \frac{R^2 - \left| x-x^0\right|^2}{\omega_{n-1} R^{n-1}} \int \limits_{S_{R} \left( x^0 \right)} \frac{\varphi(y) ds}{\left| x-x^0 -y  \right|^n}.
\end{equation}
Формула \eqref{2.5.4} даёт гармоническую в шаре $B_{R} \left( x^0 \right)$ функцию $v$, удовлетворяющую условию Дирихле $\left. v \right|_{S_{R} \left( x^0 \right)} = \varphi$. Заметим, что функция $w = u-v$ также удовлетворяет условию среднего вида \eqref{2.5.3}, но уже в шаре $B_{R} \left( x^0 \right)$. Из свойства среднего по сфере легко следует свойство среднего по шару, как в доказательстве теоремы \ref{t.2.2}

Бросим взгляд на доказательство принципа максимума \ref{t.2.3} Оно использует только то обстоятельство, что субгармоническая функция удовлетворяет неравенству со средним по шару. И больше ничего! Поэтому принцип максимума и аналогичный ему принцип минимума распространяются на все функции, удовлетворяющие условию среднего по шару. Отсюда следует, что функция $w \in C \left( \overline{B}_{R} \left( x^0 \right) \right)$ достигает свой максимум и минимум на граничной сфере $S_{R} \left( x^0 \right)$, где она равна нулю. Поэтому $w=0$, т.е. $u=v$ в $B_{R} \left( x^0\right)$, а силу произвольности шара $B_{R} \left( x^0 \right) \Subset \Omega$, $u=v$ --- гармонической функции в $\Omega$ \end{proof}

Из теоремы \ref{t.2.5.2} немедленно следует теорема о сходимости последовательности гармонических функций, аналогичная теорема К. Вейерштрасса из ТФКП, утверждающей, что равномерный пределе последовательности голоморфных функций есть голоморфная функция.

\begin{proposition}\label{p.2.5.1}
 Если последовательность гармонических в области $\Omega$ функций $\left\{ u_k \right\}_{k=1}^{\infty}$ равномерно сходится на любой компакте $K \subset \Omega$, то её предел --- гармоническая функция в $\Omega$\footnote{Теорема Вейерштрасса для гармонических функций.}.
\end{proposition}

Действительно, гармонические функции удовлетворяют для любой сферы $S_{R} \left( x^0 \right) \subset \Omega$ условию среднего \eqref{2.5.3}, в силу равномерной сходимости  возможен предельный переход в этой формуле. Предел также будет удовлетворять условию среднего, следовательно, по теореме \ref{t.2.5.2} он будет гармонической функцией.

Во втором параграфе мы определи субгармонические и супергармонические функции как, соответственно, решения неравенств $\Delta u \geqslant 0$ и $\Delta u \leqslant 0$. Было показано, что субгармонические функции удовлетворяют неравенствам со средним по сфере \eqref{2.22} и со средним по шару \eqref{2.24} (супергармонические функции удовлетворяют противоположным неравенствам). Можно забыть об операторе Лапласа и определить непрерывные суб- и супергармонические функции, положив в основу неравенства со средним значением.

\begin{definition}\label{d.2.5.1}
Функция $u \in C(\Omega)$ ($\Omega \subset \mathbb{R}^n$ --- область) называется субгармонической (супергармонической) в $\Omega$, если для любой сферы $S_{R} \left( x^0 \right) \subset \Omega$ она удовлетворяет неравенству со средним по сфере
\begin{equation}
\label{2.5.5}
u \left( x^0 \right) \leqslant (\geqslant) \frac{1}{\omega_{n-1} R^{n-1}} \int \limits_{S_{R} \left( x^0 \right)} u(x) ds.
\end{equation}
\end{definition}

Ещё раз отметим, что из неравенства \eqref{2.5.5} следует неравенство со среднем по шару. Легко видеть, что непрерывные субгармонические функции удовлетворяют принципу максимума, а непрерывные супергармонические функции --- принципу минимума.

Непрерывные субргармонические функции связаны с гармоническими функциями следующим образом. Пусть $u \in C(\Omega)$ --- субгармоническая функция, $B \Subset \Omega$ --- шар, $h \in C\left( \overline{B} \right)$ --- гармоническая функция в шаре $B$, такая, что $u \leqslant h$ на $\partial B$. Тогда $u \leqslant h$ в $B$, потому что функция $u - h$ удовлетворяет неравенству со средним по сфере, и следовательно, для неё верен принцип максимума. Справедливо и обратное утверждение: если функция $u \in C(\Omega)$ такова, что для любого шара $B \Subset \Omega$ и любой гармонической в этом шаре функции $h \in C \left( \overline{B} \right)$ из неравенства $u \leqslant h$ на $\partial B$ следует неравенство $u \leqslant h$ в $B$, то $u$ удовлетворяет неравенству со среднем по сфере \eqref{2.5.5}. В самом деле, в центре $x^0$ шара $B_{R} \left( x^0 \right)$ имеем $u \left( x^0 \right) \leqslant h \left( x^0 \right)  = \frac{1}{\omega_{n-1} R} \int_{S_{R} \left( x^0 \right)} u(x) ds$. Связь непрерывных супергармонических функций с гармоническими функциями аналогичная (сформулируйте соответствующие утверждения).

Опишем одну важную конструкцию с субргармоническими функциями --- гармоническую срезку. Пусть $u \in C(\Omega)$ --- субгармоническая функция, $B \Subset \Omega$ --- некоторый шар. Рассмотрим гармоническую в этом шаре функцию $\overline{u}$, задаваемую интегралом Пуассона \eqref{2.5.2} с граничной функцией $\varphi = \left. u \right|_{\partial B}$. Определим гармоническую срезку функции $u$ относительно шара $B$ по формуле
\begin{equation}
\label{2.5.6}
U(x) =  \left\{
\begin{aligned}
& \overline{u}(x),\, x \in B,\\
& u(x),\, x \in \Omega \backslash B.
\end{aligned}
\right.
\end{equation}
Легко видеть, что функция $U$ тоже субгармоническая. Действительно, в шаре $B$ субгармоническая функция $u \leqslant \overline{u}$, а для произвольных шара $B' \Subset \Omega$ и гармонической в этом шаре $h \in C \left( \overline{B}' \right)$, удовлетворяющей неравенству $U \leqslant h$ на $\partial B'$, $u \leqslant h$ в $B' \backslash B$ так как $u$ сначала была субгармонической функцией, а $\overline{u} \leqslant h$ в $B \cap B'$, потому что на границе этого пересечения $\overline{u} \leqslant h$. Гармоническая срезка супергармонической функции определяется аналогично и тоже супергармонична.

Наконец, легко заметить, что если $u_1,\ldots,u_N$ --- субгармонические (супергармонические) функции в области $\Omega$, то их максимум $\max \left\{ u_1,\ldots, u_N\right\}$ (их минимум $\min \left\{ u_1,\ldots, u_N\right\}$) есть субгармоническая (супергармоническая) функция в $\Omega$. Приступаем к изложению метода Перрона\footnote{Oskar Perron (1880 --- 1975)<...>} построения решения внутренней задачи Дирихле для уравнения Лапласа \eqref{2.27}. Его основная идея --- приближение к решению задачи в $\Omega \Subset \mathbb{R}^n$ непрерывными субгармоническими функциями, которые принадлежат пространству $C \left( \overline{\Omega} \right)$ и не превосходят функции $\varphi$ на $\partial \Omega$.

\begin{definition}\label{d.2.5.2}
Пусть $\Omega \Subset \mathbb{R}^n$ --- ограниченная область, $\varphi: \partial \Omega \to \mathbb{R}$ --- заданная граничная функция. Непрерывная в $\overline{\Omega}$ субгармоническая в $\Omega$ называется субрешением (нижним решением) внутренней задачи Дирихе \eqref{2.27} относительно $\varphi$, если она удовлетворяет неравенству $u \leqslant \varphi$ на $\partial \Omega$. Далее, пусть $S_{\varphi}$ --- множество всех субрешений задачи \eqref{2.27} относительно $\varphi$. Тогда функция
\begin{equation}
\label{2.5.7}
u(x) = \sup \limits_{v \in S_{\varphi}} v(x),\, x \in \Omega,
\end{equation}
называется решением Перрона внутренней задачи Дирихле \eqref{2.27} для уравнения Лапласа в области $\Omega$ относительно граничной функции $\varphi$.
\end{definition}

Отметим два простых факта о решении Перрона. Во-первых, множество $S_{\varphi}$ непусто, если функция $\varphi$ ограничена снизу, так как в этом случае существует по крайней мере одно субрешение $v \equiv \inf_{\partial \Omega} \varphi$. Во-вторых, если к тому же функция $\varphi$ ограничена сверху, то по принципу максимума любое субрешение $v \in S_{\varphi}$ также ограничено сверху величиной $\sup_{\partial \Omega} \varphi$. Таким образом, если $\varphi$ --- ограниченная на $\partial \Omega$ функция, то решение Перрона определено в любой точке $x \in \Omega$.

Но удивительным образом ограниченности функции $\varphi$ оказывается достаточно для того, чтобы решение Перрона было гармонической функцией в $\Omega$. При этом ничего не требуется от границы области, не нужна даже непрерывность функции $\varphi$. Речь идёт только о том, что функция \eqref{2.5.7} есть решение уравнения Лапласа в области $\Omega$, но говорить, что она удовлетворяет краевому условию $\left. u \right|_{\partial \Omega} = \varphi$ или служит классическим решением задачи Дирихле \eqref{2.27} ещё слишком рано.

\begin{theorem}\label{t.2.5.3}
Пусть $\Omega$ --- ограниченная область в $\mathbb{R}^n$, $\varphi: \partial \Omega \to \mathbb{R}$ --- ограниченная функция. Тогда решение Перрона \eqref{2.5.7} внутренней задачи Дирихле \eqref{2.27} относительно функции $\varphi$ есть гармоническая функция в $\Omega$.
\end{theorem}

Как это частно бывает с теоремами существования\footnote{Например, при доказательстве теоремы существования решения задачи Коши для обыкновенного дифференциального уравнения с непрерывной правой частью методом ломаных Эйлера,}, основной инструмент доказательства теоремы \ref{t.2.5.3} --- теорема Арцела. Напомним её формулировку, тем более в этой книге к теореме Арцела мы ещё будем обращаться. Пусть $K$ --- компактное метрическое пространство с функцией расстояния $d$, $C(K)$ --- векторное пространство всех непрерывных, принимающих вещественные (или комплексные) значения функций. Напомним, что $C(K)$ есть банахово пространство с равномерной нормой $\| u \|_{K} = \max_{K} |u|$.

Семейство функций $\mathcal{U} \subset C(K)$ называется предкомпактным, если его замыкание компактно. Следует отличать два определения компактности --- топологическое и секвенциальное. Семейство $\mathcal{U}$ называется: 1) топологически компактным (или просто компактным), если всякое покрытие $\mathcal{F}$ открытыми множествами допускает конечное подпокрытие; 2) секвенциально компактным, если из любой последовательности в нём можно выделить сходящуюся подпоследовательность. К счастью, для метрических пространств, в частности, для семейств $\mathcal{F} \subset C(K)$, эти определения эквивалентны, но будем пользоваться вторым.

\begin{definition}\label{d.2.5.3}
Семейство $\mathcal{U} \subset C(K)$ называется: а) равномерно ограниченным, если существует константа $C \geqslant 0$ и для любой $u \in \mathcal{U}$ $\| u\|_{K} \leqslant C$; б) равностепенно непрерывным, если для любого $\varepsilon >0$ существует $\delta > 0$, что для всех $u \in \mathcal{U}$, $x,x' \in K$ из неравенства $d \left( x,x' \right)< \delta$ следует неравенство $\left| u(x) - u(x') \right| < \varepsilon$.
\end{definition}

Вспомним, что функция непрерывная на компакте, равномерно непрерывно на нём, поэтому семейство, состоящее из конечного числа функции, равномерно ограничено и равностепенно непрерывно.

\begin{theorem}\label{t.2.5.4}
Семейство $\mathcal{U} \subset C(K)$ предкомпактно тогда и только тогда, когда оно равномерно ограниченное и равностепенно непрерывное\footnote{Теорема Арцелла или обобщённая теорема Арцелла, а иногда теорема Арцела--Аксоли.}.
\end{theorem}

Доказательство теоремы \ref{t.2.5.4}  можно найти в [] или []. Следующая лемма обобщает теорему Монтеля из ТФКП о предкомпактности семейства голоморфных функций (см, например в []).

\begin{lemma}\label{l.2.5.1}
Пусть $\mathcal{U} \subset C^{\infty}(\Omega)$ --- бесконечное семейство гармонических функций в $\Omega$, равномерно ограниченное на любом компакте $K \subset \Omega$. Тогда существует последовательность 
$\{ u_{k} \}_{k=1}^{\infty} \subset \mathcal{U}$, равномерно на любом компакте $K \subset \Omega$ сходящаяся к гармонической функции $u \in C^{\infty}(\Omega)$.
\end{lemma}

\begin{proof}
Пусть $K$ --- компакт в $\Omega$, покажем, что из равномерной ограниченности семейства $\mathcal{U}$ на $K$ следует равностепенная ограниченность этого семейства. Возьмём открытое подмножество $Q \Subset \Omega$, содержащее компакт $K$, и запишем априорную оценку первых производных функций $u \in \mathcal{U}$: для любых $x^0 \in K$, $j=1,\ldots,n$ имеем
\begin{equation}
\label{2.5.8}
\left| u_{x_j} \left( x^0 \right) \right| \leqslant \frac{n}{d} C,
\end{equation}
где $M = \sup_{u \in \mathcal{U}} \| u \|_{\overline{Q}}$, $d = \mathrm{dist} (K, \partial Q)$.
\end{proof}
